    1.  using UnityEngine;
    2.  using System.Collections;
    3.  using System.Collections.Generic;
    4.  using System;
    5.  using System.Threading;
    6.  using System.Linq;
    7.              
    8.  //Loom handles threading
    9.  public class Loom : MonoBehaviour
    10.  {
    11.      private static Loom _current;
    12.          
    13.      public static Loom Current
    14.      {
    15.          get
    16.          {
    17.              if (_current == null && Application.isPlaying)
    18.              {
    19.                  
    20.                  var g = GameObject.Find("Loom");
    21.                  if(g==null)
    22.                  {
    23.                      g = new GameObject("Loom");
    24.                      g.hideFlags = HideFlags.HideAndDontSave;
    25.                  }
    26.                  
    27.                  _current = g.GetComponent<Loom>() ?? g.AddComponent<Loom>();
    28.              }
    29.                          
    30.              return _current;
    31.          }
    32.      }
    33.      
    34.      void Awake()
    35.      {
    36.          if(_current != null && _current != this)
    37.          {
    38.              DestroyImmediate(gameObject);
    39.          }
    40.          else
    41.          {
    42.              _current = this;
    43.          }
    44.      }
    45.          
    46.      private List<Action> _actions = new List<Action>();
    47.      public class DelayedQueueItem
    48.      {
    49.          public float time;
    50.          public Action action;
    51.          public string name;
    52.      }
    53.      private List<DelayedQueueItem> _delayed = new  List<DelayedQueueItem>();
    54.                  
    55.      public static void QueueOnMainThread(Action action, float time, string name)
    56.      {
    57.          lock(Current._delayed)
    58.          {
    59.              if(Current._delayed.Any(d=>d.name==name))
    60.                  return;
    61.              QueueOnMainThread(action, time);
    62.          }
    63.      }
    64.      
    65.      public static void QueueOnMainThread(Action action, string name)
    66.      {
    67.          QueueOnMainThread(action, 0, name);
    68.      }
    69.      
    70.      public static void QueueOnMainThread(Action action, float time)
    71.      {
    72.          if (time != 0)
    73.          {
    74.              lock (Current._delayed)
    75.              {
    76.                  Current._delayed.Add(new DelayedQueueItem { time = Time.time + time, action = action});
    77.              }
    78.          }
    79.          else
    80.          {
    81.              lock (Current._actions)
    82.              {
    83.                  Current._actions.Add(action);
    84.              }
    85.          }
    86.      }
    87.      
    88.      public static void QueueOnMainThread(Action action)
    89.      {
    90.          lock (Current._actions)
    91.          {
    92.              Current._actions.Add(action);
    93.          }
    94.      }
    95.                  
    96.      public static void RunAsync(Action a)
    97.      {
    98.          var t = new Thread(RunAction);
    99.          t.Priority = System.Threading.ThreadPriority.Normal;
    100.          t.Start(a);
    101.      }
    102.                  
    103.      private static void RunAction(object action)
    104.      {
    105.          ((Action)action)();
    106.      }
    107.      
    108.                  
    109.      Action[] toBeRun = new Action[1000];
    110.      DelayedQueueItem[] toBeDelayed = new DelayedQueueItem[1000];
    111.      
    112.      void Update()
    113.      {
    114.          try
    115.          {
    116.              var actions = 0;
    117.              var delayedCount = 0;
    118.              //Process the non-delayed actions
    119.              lock (_actions)
    120.              {
    121.                  for(var i = 0; i < _actions.Count; i++)
    122.                  {
    123.                      toBeRun[actions++] = _actions[i];
    124.                      if(actions==999)
    125.                          break;
    126.                  }
    127.                  _actions.Clear();
    128.              }
    129.              for(var i = 0; i < actions; i++)
    130.              {
    131.                  var a = toBeRun[i];
    132.                  try            
    133.                  {
    134.                      a();
    135.                  }
    136.                  catch (Exception e)
    137.                  {
    138.                      Debug.LogError("Queued Exception: " + e.ToString());
    139.                  }
    140.              }
    141.              lock (_delayed)
    142.              {
    143.                  for(var i = 0; i <_delayed.Count; i++)
    144.                  {
    145.                      var d = _delayed[i];
    146.                      if(d.time < Time.time)
    147.                      {
    148.                          toBeDelayed[delayedCount++]= d;
    149.                          if(delayedCount==999)
    150.                              break;
    151.                      }
    152.                  } 
    153.              }
    154.              for(var i = 0; i < delayedCount; i++)
    155.              {
    156.                  var delayed = toBeDelayed[i];
    157.                  lock (_delayed)
    158.                  {
    159.                      _delayed.Remove(delayed);
    160.                  }
    161.                  try
    162.                  {
    163.                      delayed.action();
    164.                  }
    165.                  catch (Exception e)
    166.                  {
    167.                      Debug.LogError("Delayed Exception:" + e.ToString());
    168.                  }
    169.              }
    170.              
    171.          }
    172.          catch(Exception e)
    173.          {
    174.              Debug.LogError("Loom Error " + e.ToString());
    175.          }
    176.      }
    177.  }

    So you can put any code you like where you have [...] in your example - it can access local variables, class variables etc etc and call anything BUT nothing must touch a Unity object (read or update it in any way) or it will fail. So you can call your code to send to the server etc. Let's imagine that the server returns am X,Y,Z coordinate that you are supposed to update the game object with - you can't do that in a second thread - hence the QueueOnMainThread.

       Loom.RunAsync(()=>{
            var newPosition = SomehowGetVector3FromServer();
            Loom.QueueOnMainThread(()=>{
                    transform.position = newPosition;
            });
   });

    AcceptSocket is a blocking call - you don't want to be calling that unless you know that there is a request pending - you can tell that using this. Otherwise Unity will hang until a request appears.

    Coroutines are not threads - a coroutine that blocks will lock up Unity.
