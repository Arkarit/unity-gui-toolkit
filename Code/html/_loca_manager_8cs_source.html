85 {

89 }
90

94 // Try desired language first, without committing state yet.
95 if (ChangeLanguageImpl(_languageId))

97 CommitLanguage(_languageId, _invokeEvent);

99 }

103 // Fallback

105 if (Language == fallback)
106 {
107 // Already at fallback; still notify if requested.
108 if (_invokeEvent)

110 return false;
111 }
112
113 if (ChangeLanguageImpl(fallback))
114 {
115 CommitLanguage(fallback, _invokeEvent);
116 return false; // indicates we fell back
117 }
118
119 UiLog.LogError("Fallback language 'dev' could not be set.");
120 return false;
121 }
