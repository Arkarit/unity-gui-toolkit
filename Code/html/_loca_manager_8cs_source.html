213 private bool ChangeLanguage( string _languageId, bool _invokeEvent )
214 {
215 if (string.IsNullOrWhiteSpace(_languageId))
216 {
217 _languageId = "dev";
218 }
219
220 if (Language == _languageId)
221 return true;
222
223 // Try desired language first, without committing state yet.
224 if (ChangeLanguageImpl(_languageId))
225 {
226 CommitLanguage(_languageId, _invokeEvent);
227 return true;
228 }
229
230 UiLog.LogWarning($"Language '{_languageId}' not found. Falling back to 'dev'.");
231
232 // Fallback
233 const string fallback = "dev";
234 if (Language == fallback)
235 {
236 // Already at fallback; still notify if requested.
237 if (_invokeEvent)
238 UiEventDefinitions.EvLanguageChanged.Invoke(fallback);
239 return false;
240 }
241
242 if (ChangeLanguageImpl(fallback))
243 {
244 CommitLanguage(fallback, _invokeEvent);
245 return false; // indicates we fell back
246 }
247
248 UiLog.LogError("Fallback language 'dev' could not be set.");
249 return false;
250 }
