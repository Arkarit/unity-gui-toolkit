28 [MenuItem(StringConstants.SCENE_CHANGE_TRACKER_MENU_NAME, priority = Constants.SCENE_CHANGE_TRACKER_MENU_PRIORITY)]

34 private static bool IsEnabled
35 {
36 get => s_enabled;
37 set
38 {
39 if (value == s_enabled)
40 return;
41
42 s_enabled = value;
43
44 Menu.SetChecked(StringConstants.SCENE_CHANGE_TRACKER_MENU_NAME, s_enabled);
45 if (s_enabled)
46 {
47 Undo.postprocessModifications += OnPostProcessModifications;
48 EditorSceneManager.sceneDirtied += SceneDirtied;
49 }
50 else
51 {
52 Undo.postprocessModifications -= OnPostProcessModifications;
53 EditorSceneManager.sceneDirtied -= SceneDirtied;
54 }
55
56 EditorPrefs.SetBool(PrefsKey, s_enabled);
57 }
58 }
59
60 private static void SceneDirtied( Scene scene )
61 {
62 UiLog.LogInternal( $"sceneDirtied on {scene.name}. (See full stacktrace)" );
63 }
64
65 private static UndoPropertyModification[] OnPostProcessModifications( UndoPropertyModification[] propertyModifications )
66 {
67 foreach( UndoPropertyModification mod in propertyModifications )
68 UiLog.LogInternal(BuildLogMessage(mod));
69
70 return propertyModifications;
71 }
72
73 private static string BuildLogMessage(UndoPropertyModification mod)
74 {
75 string result = LogPrefix;
76 MonoBehaviour monoBehaviour = mod.currentValue.target as MonoBehaviour;
77 if (monoBehaviour)
78 {
79 result += $"'{monoBehaviour.gameObject.name}.{monoBehaviour.GetType().Name}.{mod.currentValue.propertyPath}'";
80 }
81 else
82 {
83 result += $"type:'{mod.currentValue.target.GetType().Name}' name:'{mod.currentValue.target.name}'";
84 }
85
86 result += $" from {mod.previousValue.value} to {mod.currentValue.value}";
87 return result;
88 }
