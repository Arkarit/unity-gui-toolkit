146 IEnumerator ChangeParentDelayed()
147 {
148 RectTransform.SetParent(null, false);
149 var activeScene = SceneManager.GetActiveScene();
150 SceneManager.MoveGameObjectToScene(gameObject, activeScene);
151 RectTransform.SetParent(m_options.Parent, false);
152 gameObject.SetActive(false);
153 yield return null;
154 if (RectTransform == null)
155 yield break;
156
157 gameObject.SetActive(true);
158 ResetTransform();
159 SetSiblingIndex();
160 }

181 private void SetSiblingIndex()
182 {
183 if (m_options.UseSiblingIndex)
184 {
185 if (RectTransform.parent == null)
186 throw new InvalidOperationException(
187 $"Cannot apply sibling index {m_options.SiblingIndex} to '{name}' because it has no parent.");
188
189 int childCount = RectTransform.parent.childCount;
190 if (childCount == 0)
191 throw new InvalidOperationException(
192 $"Cannot apply sibling index {m_options.SiblingIndex} to '{name}' because its parent '{RectTransform.parent.name}' has no children.");
193
194 int maxIndex = childCount - 1;
195 if (m_options.SiblingIndex < 0 || m_options.SiblingIndex > maxIndex)
196 throw new ArgumentOutOfRangeException(
197 nameof(m_options.SiblingIndex),
198 $"Requested sibling index {m_options.SiblingIndex} for '{name}' is out of range [0..{maxIndex}] for parent '{RectTransform.parent.name}'.");
199
200 RectTransform.SetSiblingIndex(m_options.SiblingIndex);
201 Show();
202 }

275 protected virtual void EvaluateOptions( Options _options )
276 {
277 m_options = _options;
278
279 bool closable =
280 _options.AllowOutsideTap
281 || (_options.ShowCloseButton && m_optionalCloseButton != null)
282 || _options.ButtonInfos.Length > 0;
283
284 if (!closable)
285 throw new InvalidOperationException("Dialog has no close path (no buttons, no outside tap, no X).");
286
287 if (m_maxButtons != Constants.INVALID && _options.ButtonInfos.Length > m_maxButtons)
288 UiLog.LogWarning($"Dialog '{gameObject}' contains {_options.ButtonInfos.Length} buttons; maximum supported are {m_maxButtons}. Visual problems may appear.");
289
290 m_layer = _options.Layer;
291 m_buttonContainer.SetActive(_options.ButtonInfos.Length > 0);
292 for (int i = 0; i < _options.ButtonInfos.Length; i++)
293 {
294 ButtonInfo bi = _options.ButtonInfos[i];
295 Debug.Assert(!string.IsNullOrEmpty(bi.Text) && bi.Prefab != null, $"Wrong button info number {i} - either text or prefab not set");
296 if (string.IsNullOrEmpty(bi.Text) || bi.Prefab == null)
297 continue;
298
299 UiButton button = bi.Prefab.PoolInstantiate();
300 button.transform.SetParent(m_buttonContainer.transform, false);
301 button.transform.localScale = Vector3.one * m_buttonScale;
302
303 if (!bi.EnabledInHierarchy)
304 {
305 switch (bi.DisabledButtonsHandling)
306 {
307 case EDisabledButtonsHandling.None:
308 break;
309 case EDisabledButtonsHandling.Hide:
310 button.gameObject.SetActive(false);
311 break;
312 case EDisabledButtonsHandling.DisableInHierarchy:
313 button.EnabledInHierarchy = false;
314 break;
315 case EDisabledButtonsHandling.SetNonActive:
316 button.Button.interactable = false;
317 break;
318 default:
319 throw new ArgumentOutOfRangeException();
320 }
321 }
322
323 m_buttons.Add(button);
324 m_listeners.Add(bi.OnClick);
325
326 // in a real programming language you would be able to declare the lambda capture mode (ref or copy)
327 int buttonIdx = i;
328 button.OnClick.AddListener(() =>
329 {
330 OnClick(buttonIdx);
331 });
332
333 button.Text = bi.Text;
334 }
335
336 if (m_cancelButtonsLeftSide && _options.ButtonInfos.Length > 1)
337 {
338 m_buttons[0].transform.SetAsLastSibling();
339 m_buttons[m_buttons.Count - 1].transform.SetAsFirstSibling();
340 }
341
342 if (m_optionalCloseButton)
343 {
344 m_closeButtonAction = _options.CloseButtonAction;
345 m_optionalCloseButton.OnClick.AddListener(OnCloseButton);
346 }
347
348 if (_options.AllowOutsideTap)
349 OnClickCatcher = OnCloseButton;
350 else
351 OnClickCatcher = Wiggle;
352
353 if (!m_optionalCloseButton)
354 return;
355
356 bool hasCloseButton = _options.ShowCloseButton;
357 m_optionalCloseButton.gameObject.SetActive(hasCloseButton);
358 }

363 private void HandleButtonClick( int _idx )
364 {
365 bool idxIsCloseButton = _idx == -1;
366 bool buttonClosesRequester = idxIsCloseButton || m_options.ButtonInfos[_idx].CloseRequester;
367 Debug.Assert(_idx < m_listeners.Count && _idx >= -1);
368
369 if (buttonClosesRequester)
370 {
371 if (m_consumed)
372 return;
373
374 m_consumed = true;
375 SetAllButtonsInteractability(false);
376 }
377
378 try
379 {
380 if (_idx == -1)
381 {
382 m_closeButtonAction?.Invoke();
383 }
384 else
385 {
386 if (_idx < m_listeners.Count && m_listeners[_idx] != null)
387 m_listeners[_idx]();
388 }
389 }
390 catch (Exception ex)
391 {
392 Debug.LogException(ex);
393 }
394 finally
395 {
396 if (buttonClosesRequester)
397 {
398 m_requesterHandle.MarkButton(_idx);
399 Hide(false, () => m_requesterHandle.MarkClosed());
400 }
401 }
402 }
