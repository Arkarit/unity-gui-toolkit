238 private IByteStore ResolveStore( string _key )
239 {
240 if (string.IsNullOrEmpty(_key))
241 {
242 return m_fallbackStore;
243 }
244

246 IByteStore? bestStore = null;
247
248 for (int i = 0; i < m_routes.Count; i++)
249 {
250 string prefix = m_routes[i].prefix;
251 if (_key.StartsWith(prefix, StringComparison.Ordinal))
252 {
253 if (prefix.Length > bestLen)
254 {
255 bestLen = prefix.Length;

257 }
258 }
