253 private void WireItemClick( GameObject item, int index )
254 {
255 if (m_options == null)
256 return;
257
258 bool hasClickCallback = m_options.OnItemClicked != null;
259 bool autoClose = m_options.CloseOnItemClick;
260
261 if (!hasClickCallback && !autoClose)
262 return;
263
264 // Prefer UiButton over Unity Button for consistent event handling.
265 var uiButton = item.GetComponent<UiButton>();
266 if (uiButton != null)
267 {
268 uiButton.OnClick.AddListener(() => HandleItemClick(index));
269 return;
270 }
271
272 var button = item.GetComponent<Button>();
273 if (button != null)
274 button.onClick.AddListener(() => HandleItemClick(index));

295 private void PositionAtAnchor( RectTransform anchor )
296 {
297 if (m_popupContainer == null || anchor == null)
298 return;
299
300 Canvas canvas = Canvas;
301 if (canvas == null)
302 return;
303
304 // Force layout rebuild so rect sizes are up to date.
305 Canvas.ForceUpdateCanvases();
306
307 var canvasRect = (RectTransform)canvas.transform;
308 Camera cam = canvas.renderMode == RenderMode.ScreenSpaceOverlay ? null : canvas.worldCamera;
309
310 // World corners: [0]=BL [1]=TL [2]=TR [3]=BR
311 Vector3[] corners = new Vector3[4];
312 anchor.GetWorldCorners(corners);
313
314 // Convert anchor bottom-left and top-left to popup-canvas local space.
315 RectTransformUtility.ScreenPointToLocalPointInRectangle(
316 canvasRect,
317 RectTransformUtility.WorldToScreenPoint(cam, corners[0]),
318 cam,
319 out Vector2 anchorBL);
320
321 RectTransformUtility.ScreenPointToLocalPointInRectangle(
322 canvasRect,
323 RectTransformUtility.WorldToScreenPoint(cam, corners[1]),
324 cam,
325 out Vector2 anchorTL);
326
327 // Use top-left pivot so anchoredPosition represents the popup's top-left corner.
328 // anchorMin/Max (0.5, 0.5) centres the anchor on the canvas, matching the
329 // canvas-centred local-space coordinates returned by ScreenPointToLocalPointInRectangle.
330 m_popupContainer.anchorMin = new Vector2(0.5f, 0.5f);
331 m_popupContainer.anchorMax = new Vector2(0.5f, 0.5f);
332 m_popupContainer.pivot = new Vector2(0f, 1f);
333
334 Vector2 popupSize = m_popupContainer.rect.size;
335 Rect canvasBounds = canvasRect.rect;
336
337 // Default: popup's top-left sits at anchor's bottom-left (popup below anchor).
338 Vector2 pos = anchorBL + (m_options?.Offset ?? Vector2.zero);
339
340 // Flip above anchor when the popup would overflow the canvas bottom.
341 if (pos.y - popupSize.y < canvasBounds.yMin)
342 pos.y = anchorTL.y + popupSize.y;
343
344 // Clamp X so the popup stays within the canvas width.
345 if (pos.x + popupSize.x > canvasBounds.xMax)
346 pos.x = canvasBounds.xMax - popupSize.x;
347 if (pos.x < canvasBounds.xMin)
348 pos.x = canvasBounds.xMin;
349
350 m_popupContainer.anchoredPosition = pos;
351 }
