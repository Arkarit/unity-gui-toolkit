527 _poEntry.MsgStrForms[i] = _entry.Forms[i];
528 modified = true;

619 internal static List<List<string>> ParseSheetValues(string _json)
620 {
621 var result = new List<List<string>>();
622 if (string.IsNullOrEmpty(_json))
623 return result;
624
625 int valuesIdx = _json.IndexOf("\"values\"", StringComparison.Ordinal);
626 if (valuesIdx < 0)
627 return result;
628
629 int outerStart = _json.IndexOf('[', valuesIdx);
630 if (outerStart < 0)
631 return result;
632
633 int pos = outerStart + 1;
634 int len = _json.Length;
635
636 while (pos < len)
637 {
638 SkipWhitespace(_json, ref pos);
639 if (pos >= len || _json[pos] == ']') break;
640
641 if (_json[pos] == '[')
642 {
643 pos++;
644 var row = new List<string>();
645
646 while (pos < len)
647 {
648 SkipWhitespace(_json, ref pos);
649 if (pos >= len) break;
650
651 char ch = _json[pos];
652 if (ch == ']') { pos++; break; }
653 if (ch == ',') { pos++; continue; }
654
655 if (ch == '"')
656 {
657 row.Add(ParseJsonString(_json, ref pos));
658 }
659 else
660 {
661 int end = pos;
662 while (end < len && _json[end] != ',' && _json[end] != ']') end++;
663 row.Add(_json.Substring(pos, end - pos).Trim());
664 pos = end;
665 }
666 }
667 result.Add(row);
668 }
669 else if (_json[pos] == ',')
670 {
671 pos++;
672 }
673 else
674 {
675 break;
676 }
677 }
678
679 return result;
680 }

687 private static string ParseJsonString(string _json, ref int _pos)
688 {
689 _pos++; // skip opening '"'
690 var sb = new StringBuilder();
691
692 while (_pos < _json.Length)
693 {
694 char c = _json[_pos];
695 if (c == '"') { _pos++; break; }
696
697 if (c == '\\' && _pos + 1 < _json.Length)
698 {
699 _pos++;
700 switch (_json[_pos])
701 {
702 case '"': sb.Append('"'); break;
703 case '\\': sb.Append('\\'); break;
704 case '/': sb.Append('/'); break;
705 case 'n': sb.Append('\n'); break;
706 case 'r': sb.Append('\r'); break;
707 case 't': sb.Append('\t'); break;
708 default: sb.Append(_json[_pos]); break;
709 }
710 }
711 else
712 {
713 sb.Append(c);
714 }
715 _pos++;
716 }
717 return sb.ToString();
718 }

720 private static string JsonEscape(string _s)
721 {
722 if (_s == null)
723 return string.Empty;
724
725 var sb = new StringBuilder(_s.Length + 4);
726 foreach (char c in _s)
727 {
728 switch (c)
729 {
730 case '"': sb.Append("\\\""); break;
731 case '\\': sb.Append("\\\\"); break;
732 case '\n': sb.Append("\\n"); break;
733 case '\r': sb.Append("\\r"); break;
734 case '\t': sb.Append("\\t"); break;
735 default:
736 if (c < 0x20)
737 sb.Append($"\\u{(int)c:x4}");
738 else
739 sb.Append(c);
740 break;
741 }
742 }
743 return sb.ToString();

758 internal static string ReverseKeyAffixes(string _key, string _prefix, string _postfix)
759 {
760 string result = _key ?? string.Empty;
761 string prefix = _prefix ?? string.Empty;
762 string postfix = _postfix ?? string.Empty;
763
764 if (prefix.Length > 0 && result.StartsWith(prefix, StringComparison.Ordinal))
765 result = result.Substring(prefix.Length);
766 if (postfix.Length > 0 && result.EndsWith(postfix, StringComparison.Ordinal))
767 result = result.Substring(0, result.Length - postfix.Length);
