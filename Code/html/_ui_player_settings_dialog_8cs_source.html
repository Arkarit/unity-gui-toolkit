64 private void Build()
65 {
66 ClearDialogEntries();
67
68 var categorized = m_playerSettings.GetCategorized();
69 ToggleGroup tabToggleGroup = m_tabContentContainer.GetOrCreateComponent<ToggleGroup>();
70 tabToggleGroup.allowSwitchOff = false;
71
72 foreach (var category in categorized)
73 {
74 UiTab tab = UiPool.Instance.Get(m_tabPrefab, m_tabContentContainer);
75 tab.Text = category.Key;
76 tab.Toggle.group = tabToggleGroup;
77 tab.gameObject.name = "Tab_" + category.Key;
78
79 UiPanel page = UiPool.Instance.Get(m_tabPagePrefab, m_pageContentContainer);
80 RectTransform pageContainer = page.RectTransform;
81 UiContentContainer contentContainer = pageContainer.GetComponent<UiContentContainer>();
82 if (contentContainer != null)
83 pageContainer = contentContainer.ContentContainer;
84 page.gameObject.name = "Page_" + category.Key;
85
86 m_tabInfos.Add(new TabInfo { Tab = tab, Page = page });
87
88 foreach (var group in category.Value)
89 {
90 UiTextContainer textContainer = UiPool.Instance.Get(m_groupPrefab, pageContainer);
91 textContainer.Text = group.Key;
92 textContainer.gameObject.name = "TextContainer_" + group.Key;
93
94 ToggleGroup toggleGroup = null;
95 if (group.Value.Count > 0 && group.Value[0].IsRadio)
96 {
97 toggleGroup = textContainer.GetOrCreateComponent<ToggleGroup>();
98 // We need to allow switch off while creating the entries;
99 // otherwise Unity will automatically select the first entry, which confuses
100 // all radio toggles (because they get their on/off state from player prefs)
101 toggleGroup.allowSwitchOff = true;
102 }
103
104 foreach (var entry in group.Value)
105 {
106 InstantiateMatchingEntry(entry, textContainer.RectTransform, toggleGroup);
107 }
108
109 // Only after all toggle groups elements have been added, we need to disallow switch off
110 if (toggleGroup != null)
111 toggleGroup.allowSwitchOff = false;
112 }
113 }
114
115 // After all player setting elements have been added, invoke a OnPlayerSettingChanged for each element
116 // to have a defined state, except for buttons, which haven't got a state/value
117 foreach (var kv in m_uiPlayerSettings)
118 {
119 foreach (UiPlayerSettingBase uiPlayerSetting in kv.Value)
120 if (!uiPlayerSetting.PlayerSetting.IsButton)
121 uiPlayerSetting.PlayerSetting.InvokeEvents();
122 }
123 }

125 private void InstantiateMatchingEntry( PlayerSetting _playerSetting, RectTransform _parent, ToggleGroup _toggleGroup )
126 {
127 if (_playerSetting.IsRadio)
128 {
129 foreach (string s in _playerSetting.Options.StringValues)
130 {
131 if (_playerSetting.IsLanguage)
132 InstantiateMatchingEntry(_playerSetting, m_languagePrefab, _parent, _toggleGroup, "PlayerSettingLanguage_", s);
133 else
134 InstantiateMatchingEntry(_playerSetting, m_radioPrefab, _parent, _toggleGroup, "PlayerSettingRadio_", s);
135 }
136 }
137 else if (_playerSetting.IsFloat)
138 InstantiateMatchingEntry(_playerSetting, m_sliderPrefab, _parent, null, "PlayerSettingSlider_");
139 else if (_playerSetting.IsBool)
140 InstantiateMatchingEntry(_playerSetting, m_togglePrefab, _parent, null, "PlayerSettingCheck_");
141 else if (_playerSetting.IsKeyBinding)
142 InstantiateMatchingEntry(_playerSetting, m_keyBindingsPrefab, _parent, null, "PlayerSettingKeyBinding_");
143 else if (_playerSetting.IsButton)
144 InstantiateMatchingEntry(_playerSetting, m_buttonPrefab, _parent, null, "PlayerSettingButton_");
145 else
146 {
147 var customEntry = InstantiateCustomPrefabIfNecessary(_playerSetting, _parent);
148 if (customEntry != null)
149 {
150 InstantiateMatchingEntry(_playerSetting, customEntry, _parent, null, "PlayerSetting_Custom_", null, true);
151 return;
152 }
153
154 UiLog.LogError("Unknown player setting type");
155 }
156 }

158 private UiPlayerSettingBase InstantiateCustomPrefabIfNecessary(PlayerSetting _playerSetting, RectTransform _parent)
159 {
160 if (_playerSetting.Options == null || _playerSetting.Options.CustomPrefab == null)
161 return null;
162
163 GameObject customPrefabGo = _playerSetting.Options.CustomPrefab;
164 var playerSettingBase = customPrefabGo.GetComponent<UiPlayerSettingBase>();
165 if (playerSettingBase == null)
166 {
167 UiLog.LogError($"Custom Prefab defined, but doesn't contain a '{nameof(UiPlayerSettingBase)}' component");
168 return null;
169 }
170
171 return UiPool.Instance.Get(playerSettingBase, _parent);
172 }

174 private void InstantiateMatchingEntry
175 (
176 PlayerSetting _playerSetting,
177 UiPlayerSettingBase _prefab,
178 RectTransform _parent,
179 ToggleGroup _toggleGroup,
180 string _gameObjectNamePrefix,
181 string _subKey = null,
182 bool _preInstantiated = false
183 )
184 {
185 if (!_preInstantiated)
186 {
187 var customEntry = InstantiateCustomPrefabIfNecessary(_playerSetting, _parent);
188 if (customEntry != null)
189 {
190 _prefab = customEntry;
191 _preInstantiated = true;
192 }
193 }
194
195 UiPlayerSettingBase entry = _preInstantiated ? _prefab : UiPool.Instance.Get(_prefab, _parent);
196 entry.SetData(_gameObjectNamePrefix, _playerSetting, _subKey );
197 if (_toggleGroup && entry.Toggle != null)
198 entry.Toggle.group = _toggleGroup;
199
200 List<UiPlayerSettingBase> list;
201 if (m_uiPlayerSettings.TryGetValue(_playerSetting.Key, out list))
202 {
203 list.Add(entry);
204 return;
205 }
206
207 list = new List<UiPlayerSettingBase> { entry };
208 m_uiPlayerSettings.Add(_playerSetting.Key, list);
209 }
