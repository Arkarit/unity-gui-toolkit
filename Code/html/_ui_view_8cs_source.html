214 protected virtual void OnEvFullscreenView( UiView _view, bool _show )
215 {
216 if (!m_hideOnOccludedByFullScreen)
217 return;
218
219 if (_view == this)
220 return;
221
222 // Only react if the other view's layer should occlude this view.
223 if (_view.Layer > Layer)
224 return;
225
226 // If our current visibility already matches the requested target, ignore.
227 if (_show != Visible)
228 return;
229
230 // When another fullscreen view shows, temporarily hide this one.
231 if (_show)
232 {
233 if (m_temporarilyHidden)
234 return;
235
236 m_temporarilyHidden = true;
237 m_temporarilyHiddenAutoDestroyOnHide = m_autoDestroyOnHide;
238 m_autoDestroyOnHide = false;
239 Hide(m_hideOnOccludedByFullScreenInstant);
240 return;
241 }
242
243 // When the other fullscreen view hides, restore this one.
244 if (!m_temporarilyHidden)
245 return;
246
247 m_temporarilyHidden = false;
248 m_autoDestroyOnHide = m_temporarilyHiddenAutoDestroyOnHide;
249 Show(m_hideOnOccludedByFullScreenInstant);
250 }
