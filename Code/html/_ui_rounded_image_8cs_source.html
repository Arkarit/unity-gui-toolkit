378 private void ApplyToMesh( Mesh _mesh )
379 {
380 var vertexCount = s_vertices.Count;
381
382 var vertices = new Vector3[vertexCount];
383 var colors = new Color[vertexCount];
384 var uv = new Vector2[vertexCount];
385
386 for (int i = 0; i < vertexCount; i++)
387 {
388 var vertex = s_vertices[i];
389 vertices[i] = vertex.Position;
390 colors[i] = vertex.Color;
391 uv[i] = vertex.Uv;
392 }
393
394 var triangleCount = s_triangles.Count;
395 var triangles = new int[triangleCount * 3];
396 int it = 0;
397 for (int i = 0; i < triangleCount; i++)
398 {
399 var triangle = s_triangles[i];
400 triangles[it++] = triangle[0];
401 triangles[it++] = triangle[1];
402 triangles[it++] = triangle[2];
403 }
404
405 _mesh.vertices = vertices;
406 _mesh.colors = colors;
407 _mesh.uv = uv;
408 _mesh.triangles = triangles;
409
410 s_vertices.Clear();
411 s_triangles.Clear();
412 }

533 {
534 var x = _rect.x;
535 var y = _rect.y;
536 var w = _rect.width;
537 var h = _rect.height;
538
539 Rect l = new Rect(x, y + _radius, _frameSize, h - _radius * 2);
540 Rect r = new Rect(w + x - _frameSize, y + _radius, _frameSize, h - _radius * 2);
541 Rect t = new Rect(x + _radius, h + y - _frameSize, w - _radius * 2, _frameSize);

544 switch (_fade)

566 }

568 AddFrameSegment(_rect, Corner.TopLeft, _frameSize, _radius, _fade);
569 AddFrameSegment(_rect, Corner.TopRight, _frameSize, _radius, _fade);

592 private void FadeFrameRect( Rect _rect, Fade _fade )
593 {
594 if (_fade == Fade.None)
595 return;
596
597 float top = _rect.yMin;
598 float bottom = _rect.yMax;
599 float left = _rect.xMin;
600 float right = _rect.xMax;
601
602 // frame is 8 quads, 16 tris, 32 verts
603 for (int i = s_vertices.Count - 32; i < s_vertices.Count; i++)
604 {
605 var vertex = s_vertices[i];
606 var position = vertex.Position;
607
608 bool condition =
609 Mathf.Approximately(left, position.x) ||
610 Mathf.Approximately(right, position.x) ||
611 Mathf.Approximately(top, position.y) ||
612 Mathf.Approximately(bottom, position.y);
613
614 if (_fade == Fade.Inner)
615 condition = !condition;
616
617 if (condition)
618 {
619 vertex.Color = m_fadeColor;
620 }
621 }

669 private void AddFrameSegment( Rect _rect, Corner _corner, float _frameSize, float _radius, Fade _fade )
670 {
671 var x = _rect.x;
672 var y = _rect.y;
673 var w = _rect.width;
674 var h = _rect.height;
675
676 float angle = ((int)_corner + 3) * 90 * Mathf.Deg2Rad;
677 float angleIncrement = 90f / m_cornerSegments * Mathf.Deg2Rad;
678
679 float ox, oy;
680 switch (_corner)
681 {
682 case Corner.TopLeft:
683 ox = x + _radius;
684 oy = y + h - _radius;
685 break;
686 case Corner.TopRight:
687 ox = x + w - _radius;
688 oy = y + h - _radius;
689 break;
690 case Corner.BottomRight:
691 ox = x + w - _radius;
692 oy = y + _radius;
693 break;
694 case Corner.BottomLeft:
695 ox = x + _radius;
696 oy = y + _radius;
697 break;
698 default:
699 throw new ArgumentOutOfRangeException(nameof(_corner), _corner, null);
700 }
701
702 float radiusInner = _radius - _frameSize;
703 for (int i = 0; i < m_cornerSegments; i++)
704 {
705 float x1 = Mathf.Sin(angle) * _radius + ox;
706 float y1 = Mathf.Cos(angle) * _radius + oy;
707 float x3 = Mathf.Sin(angle) * radiusInner + ox;
708 float y3 = Mathf.Cos(angle) * radiusInner + oy;
709 angle += angleIncrement;
710 float x0 = Mathf.Sin(angle) * _radius + ox;
711 float y0 = Mathf.Cos(angle) * _radius + oy;
712 float x2 = Mathf.Sin(angle) * radiusInner + ox;
713 float y2 = Mathf.Cos(angle) * radiusInner + oy;
714 AddIrregularQuad(x0, y0, x1, y1, x2, y2, x3, y3, _fade);
715 }
716
717 }

778 private void AddQuad( Vector2 _posMin, Vector2 _posMax, QuadFade _fade = QuadFade.None, bool _left = false )
779 {
780 int startIndex = s_vertices.Count;
781
782 switch (_fade)
783 {
784 case QuadFade.None:
785 AddVert(_posMin.x, _posMin.y, color);
786 AddVert(_posMin.x, _posMax.y, color);
787 AddVert(_posMax.x, _posMax.y, color);
788 AddVert(_posMax.x, _posMin.y, color);
789 break;
790 case QuadFade.Left:
791 AddVert(_posMin.x, _posMin.y, m_fadeColor);
792 AddVert(_posMin.x, _posMax.y, m_fadeColor);
793 AddVert(_posMax.x, _posMax.y, color);
794 AddVert(_posMax.x, _posMin.y, color);
795 break;
796 case QuadFade.Right:
797 AddVert(_posMin.x, _posMin.y, color);
798 AddVert(_posMin.x, _posMax.y, color);
799 AddVert(_posMax.x, _posMax.y, m_fadeColor);
800 AddVert(_posMax.x, _posMin.y, m_fadeColor);
801 break;
802 case QuadFade.Top:
803 AddVert(_posMin.x, _posMin.y, color);
804 AddVert(_posMin.x, _posMax.y, m_fadeColor);
805 AddVert(_posMax.x, _posMax.y, m_fadeColor);
806 AddVert(_posMax.x, _posMin.y, color);
