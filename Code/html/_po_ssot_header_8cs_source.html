73 public static PoSsotInfo ParseHeader(string _filePath)
74 {
75 if (!HasSsotHeader(_filePath))
76 return null;
77
78 try
79 {
80 var info = new PoSsotInfo();
81 foreach (var line in File.ReadLines(_filePath, Encoding.UTF8))
82 {
83 if (!line.StartsWith("#"))
84 break;
85
86 if (line.StartsWith(HEADER_BRIDGE_PREFIX))
87 {
88 string rest = line.Substring(HEADER_BRIDGE_PREFIX.Length);
89 int guidOpen = rest.IndexOf(GUID_OPEN, StringComparison.Ordinal);
90 if (guidOpen >= 0)
91 {
92 info.BridgeName = rest.Substring(0, guidOpen).Trim();
93 int guidClose = rest.IndexOf(GUID_CLOSE, guidOpen + GUID_OPEN.Length, StringComparison.Ordinal);
94 if (guidClose > guidOpen)
95 info.BridgeGuid = rest.Substring(guidOpen + GUID_OPEN.Length, guidClose - guidOpen - GUID_OPEN.Length).Trim();
96 }
97 else
98 {
99 info.BridgeName = rest.Trim();
100 }
101 }
102 else if (line.StartsWith(HEADER_SOURCE_PREFIX))
103 {
104 info.SourceUrl = line.Substring(HEADER_SOURCE_PREFIX.Length).Trim();
105 }
106 else if (line.StartsWith(HEADER_GENERATED_PREFIX))
107 {
108 string ts = line.Substring(HEADER_GENERATED_PREFIX.Length).Trim();
109 if (DateTime.TryParse(ts, null, System.Globalization.DateTimeStyles.RoundtripKind, out var dt))
110 info.GeneratedAt = dt;
111 }
112 }
113 return info;
114 }
115 catch (Exception e)
116 {
117 UiLog.LogWarning($"PoSsotHeader: failed to parse header in '{_filePath}': {e.Message}");
118 return null;
119 }
120 }

182 public static List<string> StripSsotLines(List<string> _lines)
183 {
184 var result = new List<string>();
185 foreach (var line in _lines)
186 {
187 if (line.StartsWith(HEADER_LINE_1, StringComparison.Ordinal)) continue;
188 if (line.StartsWith(HEADER_BRIDGE_PREFIX,StringComparison.Ordinal)) continue;
189 if (line.StartsWith(HEADER_SOURCE_PREFIX,StringComparison.Ordinal)) continue;
190 if (line.StartsWith(HEADER_GENERATED_PREFIX, StringComparison.Ordinal)) continue;
191 if (line.StartsWith(HEADER_WARNING, StringComparison.Ordinal)) continue;
192 result.Add(line);
193 }
