106 private void CalcOffsetsIfNecessary()
107 {
108 if (m_lastBounding == Bounding &&
109 Mathf.Approximately(m_lastAngleHorizontal, m_angleHorizontal) &&
110 Mathf.Approximately(m_lastAngleVertical, m_angleVertical))
111 return;
112
113 m_lastBounding = Bounding;
114 m_lastAngleHorizontal = m_angleHorizontal;
115 m_lastAngleVertical = m_angleVertical;
116
117 var hor = Calc(m_lastBounding.height, m_angleHorizontal);
118 m_topLeft.x = Used(CornerFlags.TopLeft) ? hor : 0;
119 m_topRight.x = Used(CornerFlags.TopRight) ? hor : 0;
120 m_bottomLeft.x = Used(CornerFlags.BottomLeft) ? -hor : 0;
121 m_bottomRight.x = Used(CornerFlags.BottomRight) ? -hor : 0;
122 var vert = Calc(m_lastBounding.width, m_angleVertical);
123 m_topLeft.y = Used(CornerFlags.TopLeft) ? -vert : 0;
124 m_topRight.y = Used(CornerFlags.TopRight) ? vert : 0;
125 m_bottomLeft.y = Used(CornerFlags.BottomLeft) ? -vert : 0;
