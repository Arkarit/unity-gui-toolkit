187 private void CloneStyleConfig<T>(ref T currentStyleConfig, string _memberName, string _name) where T : UiStyleConfig
188 {
189 string resourceDir = "Assets/Resources";
190 EditorFileUtility.EnsureUnityFolderExists(resourceDir);
191 var newConfigPath = $"{resourceDir}/{_name}.asset";
192 if (File.Exists(EditorFileUtility.GetNativePath(newConfigPath)))
193 if (!EditorUtility.DisplayDialog("Overwrite Configuration?", $"A config file at '{newConfigPath}' already exists. Should it be overwritten? (Not undoable)", "OK", "Cancel"))
194 return;
195
196 currentStyleConfig = Instantiate(currentStyleConfig);
197 AssetDatabase.CreateAsset(currentStyleConfig, newConfigPath);
198 var styleConfigProp = m_serializedSettingsObject.FindProperty(_memberName);
199 styleConfigProp.objectReferenceValue = currentStyleConfig;
200 m_serializedSettingsObject.ApplyModifiedProperties();
201 AssetDatabase.SaveAssets();
202 }
