177

179 return;
180
181 Vector4 scale = Vector4.one;
182 scale.x = rect.width / m_originalBounds.size.x;
183 scale.y = rect.height / m_originalBounds.size.y;
184 switch (m_zSize)
185 {
186 case EZSize.Untouched:
187 break;
188 case EZSize.ByX:
189 scale.z = scale.x;
190 break;
191 case EZSize.ByY:
192 scale.z = scale.y;
193 break;
194 case EZSize.ByXYAverage:

196 break;

203

205 offset.Scale( scale );

208 offset.z = offset.w = 0;
