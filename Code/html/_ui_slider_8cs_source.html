155 public static string DefaultValueToStringFn(UiSlider _slider)
156 {
157 var val = _slider.TextBaseValue + _slider.Value * _slider.TextMultiplier;
158 if (_slider.TextIsInt)
159 {
160 int intVal = (int) Mathf.Round(val);
161 return intVal.ToString();
162 }
163
164 return string.Format(CultureInfo.InvariantCulture, "{0:F2}", val);
165 }

233 private void ChangeText(float _value)
234 {
235 if ( m_optionalValueText == null || TextMode == ETextMode.NoText)
236 return;
237
238 m_optionalValueText.text = ValueToStringFn(this);
239
240 if (TextMode == ETextMode.Always)
241 return;
242
243 if (m_hideTextCoroutine != null)
244 {
245 StopCoroutine(m_hideTextCoroutine);
246 m_hideTextCoroutine = StartCoroutine(HideTextDelayed());
247 return;
248 }
249
250 m_optionalValueText.gameObject.SetActive(true);
251 if (m_optionalShowTextAnimation != null)
252 m_optionalShowTextAnimation.Play();
253
254 m_hideTextCoroutine = StartCoroutine(HideTextDelayed());
255 }

257 IEnumerator HideTextDelayed()
