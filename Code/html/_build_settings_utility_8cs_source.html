60 static private bool QueryBuildSettingsStatus()
61 {
62 // If no version control provider, assume not readonly
63 if (UnityEditor.VersionControl.Provider.enabled == false)
64 return false;
65
66 // If we cannot checkout, then assume we are not readonly
67 if (UnityEditor.VersionControl.Provider.hasCheckoutSupport == false)
68 return false;
69
71 //if (UnityEditor.VersionControl.Provider.onlineState == UnityEditor.VersionControl.OnlineState.Offline)
72 // return true;
73
74 // Try to get status for file
75 var status = UnityEditor.VersionControl.Provider.Status("ProjectSettings/EditorBuildSettings.asset", false);
76 status.Wait();
77
78 // If no status listed we can edit
79 if (status.assetList == null || status.assetList.Count != 1)
80 return true;
81
82 // If is checked out, we can edit
83 if (status.assetList[0].IsState(UnityEditor.VersionControl.Asset.States.CheckedOutLocal))
84 return false;
85
86 return true;
87 }

126 static public BuildScene GetBuildScene( UnityEngine.Object sceneObject )
127 {
128 BuildScene entry = new BuildScene()
129 {
130 buildIndex = SCENE_NOT_IN_BUILD,
131 assetGUID = new GUID(string.Empty)
132 };
133
134 if (sceneObject as SceneAsset == null)
135 return entry;
136
137 entry.assetPath = AssetDatabase.GetAssetPath(sceneObject);
138 entry.assetGUID = new GUID(AssetDatabase.AssetPathToGUID(entry.assetPath));
139 Scene scene = EditorSceneManager.GetSceneByPath(entry.assetPath);
140 entry.loadedInEditor = scene != null && scene.isLoaded;
141 entry.enabled = !Application.isPlaying && (!entry.loadedInEditor || EditorSceneManager.sceneCount > 1);
142
143 for (int index = 0, sceneIndex = 0; index < EditorBuildSettings.scenes.Length; ++index)
144 {
145 if (entry.assetGUID.Equals(EditorBuildSettings.scenes[index].guid))
146 {
147 entry.scene = EditorBuildSettings.scenes[index];
148 entry.buildIndex = entry.scene.enabled ? sceneIndex : SCENE_DISABLED;
149 return entry;
150 }
151
152 EditorBuildSettingsScene sc = EditorBuildSettings.scenes[index];
153 if (sc.enabled)
154 sceneIndex++;
155 }

239 }
240
241 switch (selection)

246 onlyDisable = true;
247 break;
248 default:
249 case 2: // cancel
250 return;
251 }
252 }
253
254 // User chose to not remove, only disable the scene
255 if (onlyDisable)

257 SetBuildSceneState(buildScene, false);
258 }
259 // User chose to fully remove the scene from build settings
260 else
261 {
262 List<EditorBuildSettingsScene> tempScenes = EditorBuildSettings.scenes.ToList();
263 tempScenes.RemoveAll(scene => scene.guid.Equals(buildScene.assetGUID));
264 EditorBuildSettings.scenes = tempScenes.ToArray();
