113 private void OnGUI()
114 {
115 Event e = Event.current;
116 bool isSingleKey = m_playerSettingOptions.KeyPolicy == EKeyPolicy.SingleKey;
117
118 KeyCode keyCode = UiUtility.EventToKeyCode(e, true);
119 if (m_onEvent == null || keyCode == KeyCode.None || (GeneralUtility.IsModifierKey(keyCode) && !isSingleKey))
120 return;
121
122 KeyBinding.EModifiers modifiers = GetCurrentModifiers();
123 KeyBinding keyBinding = new KeyBinding(keyCode, modifiers);
124
125 if (m_playerSettingOptions != null && isSingleKey)
126 keyBinding = new KeyBinding(keyCode);
127
128 bool isSuppressed = IsSuppressed(keyBinding);
129 if (isSuppressed)
130 {
131 UiLog.LogInternal($"Key '{FormatKeyBindingForLog(keyBinding)}' not supported");
132
133 if (m_wiggleAnimation)
134 m_wiggleAnimation.Play();
135
136 if (m_textFieldNotSupportedWarning)
137 m_textFieldNotSupportedWarning.text =

139
140 if (m_warningAnimation)
141 m_warningAnimation.Play();
142
143 m_isClosable = false;
144 return;
145 }
146
147 if (m_wiggleAnimation)
148 m_wiggleAnimation.Reset();
149
150 if (keyCode != KeyCode.Escape)
151 m_onEvent.Invoke(keyBinding);

153 m_onEvent = null;
154
155 if (!UiUtility.IsMouse(keyCode))

177 private bool IsSuppressed( KeyBinding _keyBinding )
178 {
179 if (m_playerSettingOptions == null)
180 return false;
181
182 if (_keyBinding.KeyCode == KeyCode.Escape)
183 return false;
184
185 List<KeyCode> filterList = m_playerSettingOptions.KeyCodeFilterList;
186 if (filterList == null)
187 return false;
188
189 bool isWhiteList = m_playerSettingOptions.KeyCodeFilterListIsWhitelist;
190
191 bool containsKeyCode = filterList.Contains(_keyBinding.KeyCode);
192 return isWhiteList ? !containsKeyCode : containsKeyCode;

195 private KeyBinding.EModifiers GetCurrentModifiers()
196 {
197 KeyBinding.EModifiers mods = KeyBinding.EModifiers.None;
198
199 if (m_playerSettingOptions.KeyPolicy == EKeyPolicy.SingleKey)
200 return mods;
201
202 if (Input.GetKey(KeyCode.LeftShift) || Input.GetKey(KeyCode.RightShift))
203 mods |= KeyBinding.EModifiers.Shift;
204
205 if (Input.GetKey(KeyCode.LeftControl) || Input.GetKey(KeyCode.RightControl))
206 mods |= KeyBinding.EModifiers.Ctrl;
207
208 if (Input.GetKey(KeyCode.LeftAlt) || Input.GetKey(KeyCode.RightAlt))
209 mods |= KeyBinding.EModifiers.Alt;
210
211 return mods;
