55 private void OnTMProTextChanged( UnityEngine.Object _obj )
56 {
57 // return if the call does not apply to our TextMeshPro instance (or null or inactive)
58 if ( m_textMeshPro != _obj || this == null || !IsActive() )
59 return;
60
61 // If no text at all, return
62 var textInfo = m_textMeshPro.textInfo;
63 if (textInfo.characterCount - textInfo.spaceCount <= 0)
64 {
65 return;
66 }
67
68 // Determine, if any modifier changes the mesh topology
69 BaseMeshEffectTMP[] mods = GetComponents<BaseMeshEffectTMP>();
70 bool anyTopologyChangingMod = false;
71 foreach(var mod in mods)
72 {
73 if (mod.IsActive() && mod.ChangesTopology)
74 {
75 anyTopologyChangingMod = true;
76 break;
77 }
78 }
79
80 // Get meshes from TextMeshPro
81 s_meshes.Clear();
82 foreach (var info in textInfo.meshInfo)
83 {
84 if (info.mesh == null)
85 return;
86
87 // clone mesh if any modifier changes the topology (see "How it works")
88 Mesh mesh = anyTopologyChangingMod ? Mesh.Instantiate(info.mesh) : info.mesh;
89 s_meshes.Add(mesh);
90 }
91
92 // Fill the vertex helper, then call ModifyMesh() for all modifiers
93 int numMeshes = s_meshes.Count;
94 for(int i=0; i<numMeshes; i++)
95 {
96 Mesh m = s_meshes[i];
97 FixMissingChannelsIfNecessary(ref m);
98 VertexHelper vertexHelper = new VertexHelper(m);
99
100 // We don't check 'enabled' for the mods here - BaseMeshEffect itself also doesn't.
101 // Modifiers itself are responsible to return when inactive.
102 foreach (var mod in mods)
103 mod.ModifyMesh(vertexHelper);
104
105 vertexHelper.FillMesh(m);
106 vertexHelper.Dispose();
107 }
108
109 // Assign the mesh to the canvas renderer
110 if (m_canvasRenderer)
111 {
112 if (IsBoundsValid(s_meshes[0]))
113 m_canvasRenderer.SetMesh(s_meshes[0]);
114 GetComponentsInChildren(false, s_subMeshUIs);
115 int numModifiedMeshes = s_meshes.Count;
116 int numSubMeshes = s_subMeshUIs.Count;
117 Debug.Assert(numModifiedMeshes == numSubMeshes+1);
118 if (numModifiedMeshes == numSubMeshes+1)
119 {
120 for (int i=0; i<numSubMeshes; i++)
121 {
122 TMP_SubMeshUI subMeshUI = s_subMeshUIs[i];
123 if (IsBoundsValid(s_meshes[i+1]))
124 subMeshUI.canvasRenderer.SetMesh(s_meshes[i+1]);
125 }
126 }
127
128 s_subMeshUIs.Clear();
129 }
130
131 s_meshes.Clear();
132 }

235 private void FixMissingChannelsIfNecessary( ref Mesh _mesh )
236 {
237 int numVerts = _mesh.vertices.Length;
238 if ( numVerts == 0)

240
241 if (_mesh.uv3 == null || _mesh.uv3.Length != numVerts)
242 _mesh.uv3 = new Vector2[numVerts];

250 private void UpdateTMPCallback()
251 {
252 if (!m_textMeshPro)
253 return;
254
255 BaseMeshEffectTMP[] mods = GetComponents<BaseMeshEffectTMP>();
256
257 foreach (BaseMeshEffectTMP mod in mods)
258 mod.RemoveTMPCallback();
259
260 for (int i=0; i<mods.Length; i++)
261 {
262 BaseMeshEffectTMP mod = mods[i];
263 if (mod.enabled && !mod.m_aboutToBeDisabled)

265 mod.InstallTMPCallback();
266 return;
