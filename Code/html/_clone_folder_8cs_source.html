91 public static bool Clone(string srcFolder, string destFolder)
92 {
93 try
94 {
95 var guidRemap = CopyFolderRecursive(srcFolder, destFolder);

116 public static void RenameMatchingAssets(string folder, string oldName, string newName)
117 {
118 string[] guids = AssetDatabase.FindAssets("", new[] { folder });
119 foreach (string guid in guids)
120 {
121 string path = AssetDatabase.GUIDToAssetPath(guid);
122 if (AssetDatabase.IsValidFolder(path))
123 continue;
124
125 string baseName = Path.GetFileNameWithoutExtension(path);
126 if (!baseName.Contains(oldName))
127 continue;
128
129 string renamed = baseName.Replace(oldName, newName);
130 string error = AssetDatabase.RenameAsset(path, renamed);
131 if (!string.IsNullOrEmpty(error))
132 Debug.LogWarning($"[CloneFolder] Could not rename '{path}': {error}");
133 }
134
135 AssetDatabase.Refresh(ImportAssetOptions.ForceSynchronousImport);
136 }

147 private static Dictionary<string, string> CopyFolderRecursive(string src, string dest)
148 {
149 // Collect all file assets under source (FindAssets returns GUIDs)
150 string[] guids = AssetDatabase.FindAssets("", new[] { src });
151
152 var guidRemap = new Dictionary<string, string>(guids.Length);
153
154 // First pass: create all necessary sub-folders top-down, then copy files
155 int total = guids.Length;
156 for (int i = 0; i < total; i++)
157 {
158 string origGuid = guids[i];
159 string origPath = AssetDatabase.GUIDToAssetPath(origGuid);
160
161 // Skip if this is a folder entry itself (FindAssets can return folder GUIDs)
162 if (AssetDatabase.IsValidFolder(origPath))
163 continue;
164
165 // Build destination path by replacing the source prefix
166 string clonePath = dest + origPath.Substring(src.Length);
167
168 // Ensure parent directory exists
169 string cloneDir = Path.GetDirectoryName(clonePath)?.Replace('\\', '/');
170 if (!string.IsNullOrEmpty(cloneDir))
171 EditorFileUtility.EnsureUnityFolderExists(cloneDir);
172
173 if (EditorUtility.DisplayCancelableProgressBar(
174 "Clone Folder – Copying",
175 Path.GetFileName(origPath),
176 (float)i / total))
177 {
178 // User cancelled – undo by deleting whatever we created
179 if (AssetDatabase.IsValidFolder(dest))
180 AssetDatabase.DeleteAsset(dest);
181 return null;
182 }
183
184 if (!AssetDatabase.CopyAsset(origPath, clonePath))
185 {
186 Debug.LogWarning($"[CloneFolder] Failed to copy '{origPath}' → '{clonePath}'");
187 continue;
188 }
189
190 AssetDatabase.ImportAsset(clonePath, ImportAssetOptions.ForceSynchronousImport);
191
192 string newGuid = AssetDatabase.AssetPathToGUID(clonePath);
193 if (!string.IsNullOrEmpty(newGuid))
194 guidRemap[origGuid] = newGuid;
195 }
196
197 return guidRemap;
198 }

204 private static void RewireInternalReferences(string destFolder, Dictionary<string, string> guidRemap)
205 {
206 string[] guids = AssetDatabase.FindAssets("", new[] { destFolder });
207 int total = guids.Length;
208
209 for (int i = 0; i < total; i++)
210 {
211 string assetPath = AssetDatabase.GUIDToAssetPath(guids[i]);
212
213 // Skip folders and scripts (scripts are text; rewiring them would corrupt code)
214 if (AssetDatabase.IsValidFolder(assetPath))
215 continue;
216 if (assetPath.EndsWith(".cs") || assetPath.EndsWith(".js") || assetPath.EndsWith(".shader"))
217 continue;
218
219 if (EditorUtility.DisplayCancelableProgressBar(
220 "Clone Folder – Rewiring",
221 Path.GetFileName(assetPath),
222 (float)i / total))
223 break; // keep copies, just stop rewiring
224
225 Object[] assets = AssetDatabase.LoadAllAssetsAtPath(assetPath);
226 if (assets == null)
227 continue;
228
229 foreach (Object asset in assets)
230 {
231 if (asset == null)
232 continue;
233
234 var so = new SerializedObject(asset);
235 SerializedProperty prop = so.GetIterator();
236 bool changed = false;
237
238 while (prop.Next(true))
239 {
240 if (prop.propertyType != SerializedPropertyType.ObjectReference)
241 continue;
242
243 Object referenced = prop.objectReferenceValue;
244 if (referenced == null)
245 continue;
246
247 // Get the GUID of the referenced asset
248 if (!AssetDatabase.TryGetGUIDAndLocalFileIdentifier(referenced, out string refGuid, out long localId))
249 continue;
250
251 if (!guidRemap.TryGetValue(refGuid, out string newGuid))
252 continue;
253
254 // Load the cloned counterpart
255 string newPath = AssetDatabase.GUIDToAssetPath(newGuid);
256 Object remapped = FindAssetWithLocalId(newPath, referenced, localId);
257 if (remapped == null)
258 continue;
259
260 prop.objectReferenceValue = remapped;
261 changed = true;
262 }
263
264 if (changed)
265 {
266 so.ApplyModifiedPropertiesWithoutUndo();
267 EditorUtility.SetDirty(asset); // ensures prefab sub-objects are flushed by SaveAssets
268 }
269 }
270 }

281 private static Object FindAssetWithLocalId(string path, Object fallback, long localId)
282 {
283 Object[] all = AssetDatabase.LoadAllAssetsAtPath(path);
284 if (all == null || all.Length == 0)
285 return null;
286
287 foreach (Object a in all)
288 {
289 if (a == null)
290 continue;
291 if (AssetDatabase.TryGetGUIDAndLocalFileIdentifier(a, out _, out long id) && id == localId)
292 return a;
293 }
294
295 // Sub-asset not found by ID – return main asset (covers renamed sub-assets)
296 return all[0];
297 }

300 private static string GetSelectedFolderPath()
301 {
302 // activeObject is null when right-clicking the tree view without a prior left-click.
303 // Fall through all available selection APIs as fallbacks.
304 Object obj = Selection.activeObject;
305
306 if (obj == null)
307 {
308 var guids = Selection.assetGUIDs;
309 if (guids != null && guids.Length > 0)
310 obj = AssetDatabase.LoadAssetAtPath<Object>(AssetDatabase.GUIDToAssetPath(guids[0]));
311 }
312
313 if (obj == null && Selection.objects != null && Selection.objects.Length > 0)
314 obj = Selection.objects[0];
315
316 if (obj == null)
317 return null;
318
319 string path = AssetDatabase.GetAssetPath(obj);
320 return AssetDatabase.IsValidFolder(path) ? path : null;
321 }
