125 {
126 if (string.IsNullOrEmpty(name)) return "";

133 private static (TMP_FontAsset font, Material mat) FindMatchingTMPFontAndMaterial( string legacyFontName )
134 {
135 if (string.IsNullOrEmpty(legacyFontName))
136 return (null, null);
137
138 if (s_tmpFontLookupCache.TryGetValue(legacyFontName, out var cached))
139 return cached;
140
141 var legacyKey = NormalizeFontKey(legacyFontName);
142
143 TMP_FontAsset best = null;
144 int bestScore = int.MinValue;
145
146 // 1) scan all TMP_FontAsset assets once
147 var guids = AssetDatabase.FindAssets("t:TMP_FontAsset");
148 foreach (var guid in guids)
149 {
150 var path = AssetDatabase.GUIDToAssetPath(guid);
151 var fa = AssetDatabase.LoadAssetAtPath<TMP_FontAsset>(path);
152 if (!fa) continue;
153
154 // candidates to compare
155 var cand = new[]
156 {
157 fa.faceInfo.familyName, // most robust match
158 fa.sourceFontFile ? fa.sourceFontFile.name : null,
159 fa.name
160 };
161
162 int scoreHere = int.MinValue;
163 for (int i = 0; i < cand.Length; i++)
164 {
165 var c = cand[i];
166 if (string.IsNullOrEmpty(c)) continue;
167
168 var key = NormalizeFontKey(c);
169
170 // scoring: exact normalized match >> startswith/contains
171 if (key == legacyKey) scoreHere = Math.Max(scoreHere, 100 - i); // prefer earlier fields
172 else if (key.StartsWith(legacyKey)) scoreHere = Math.Max(scoreHere, 60 - i);
173 else if (legacyKey.StartsWith(key)) scoreHere = Math.Max(scoreHere, 50 - i);
174 else if (key.Contains(legacyKey)) scoreHere = Math.Max(scoreHere, 40 - i);
175 }
176
177 if (scoreHere > bestScore)
178 {
179 bestScore = scoreHere;
180 best = fa;
181 }
182 }
183
184 // 2) prefer the font asset's own (default) material; override only if a clearly better neutral preset exists
185 Material mat = null;
186
187 if (best)
188 {
189 // always start with the asset's own material (usually effect-free)
190 var defaultMat = best.material;
191 mat = defaultMat;
192
193 // search for neutral presets in the same folder; avoid effecty ones like Outline/Shadow/Glow/Underlay
194 var folder = System.IO.Path.GetDirectoryName(AssetDatabase.GetAssetPath(best)).Replace("\\", "/");
195 if (!string.IsNullOrEmpty(folder))
196 {
197 var matGuids = AssetDatabase.FindAssets("t:Material", new[] { folder });
198
199 // scoring helpers
200 int ScoreMaterialName( string name, IEnumerable<string> keys )
201 {
202 if (string.IsNullOrEmpty(name)) return int.MinValue;
203 var mk = NormalizeFontKey(name);
204
205 // hard penalties for effect keywords
206 // note: keep this set conservative; fontStyle handles bold/italic already
207 string[] effectHints = { "outline", "shadow", "glow", "underlay", "stroke", "soft", "thick", "thin" };
208 foreach (var eh in effectHints)
209 if (mk.Contains(NormalizeFontKey(eh))) return -1000;
210
211 int score = 0;
212
213 // small bonuses for neutral keywords
214 string[] neutralHints = { "regular", "normal", "default", "atlas", "sdf" };
215 foreach (var nh in neutralHints)
216 if (mk.Contains(NormalizeFontKey(nh))) score += 5;
217
218 foreach (var k in keys)
219 {
220 if (string.IsNullOrEmpty(k)) continue;
221 if (mk == k) score = Math.Max(score, score + 100);
222 else if (mk.StartsWith(k)) score = Math.Max(score, score + 70);
223 else if (mk.Contains(k)) score = Math.Max(score, score + 50);
224 }
225
226 return score;
227 }
228
229 // keys representing this font
230 var keysToTry = new[]
231 {
232 NormalizeFontKey(best.faceInfo.familyName),
233 NormalizeFontKey(best.name),
234 NormalizeFontKey(best.sourceFontFile ? best.sourceFontFile.name : null)
235 };
236
237 var bestPreset = defaultMat;
238 int bestPresetScore = -1;
239
240 foreach (var mg in matGuids)
241 {
242 var mPath = AssetDatabase.GUIDToAssetPath(mg);
243 var m = AssetDatabase.LoadAssetAtPath<Material>(mPath);
244 if (!m || m == defaultMat) continue;
245
246 // consider only TMP-compatible shaders
247 var shaderName = m.shader ? m.shader.name : "";
248 if (string.IsNullOrEmpty(shaderName) || (!shaderName.Contains("TextMeshPro") && !shaderName.Contains("TMP")))
249 continue;
250
251 int score = ScoreMaterialName(m.name, keysToTry);
252 if (score > bestPresetScore)
253 {
254 bestPresetScore = score;
255 bestPreset = m;
256 }
257 }
258
259 // Only override the default if the preset is clearly better AND not an "effect" (penalties enforce that)
260 // Threshold keeps us conservative; tweak if you like.
261 const int OVERRIDE_THRESHOLD = 95;
262 if (bestPreset != null && bestPreset != defaultMat && bestPresetScore >= OVERRIDE_THRESHOLD)
263 mat = bestPreset;
264 else
265 mat = defaultMat; // stick to the asset's own material
266 }
267 }
268
269 var result = (best, mat);
270 s_tmpFontLookupCache[legacyFontName] = result;
271 return result;
272 }

408 }

573 Action<TSnapshot, TB> _apply

614 UiLog.LogError(s, oldComp);

618 continue;
619 }
620

622 Undo.RegisterCompleteObjectUndo(go, "Remove Source Component");
623 Undo.DestroyObjectImmediate(oldComp);
624
625 // 3) Ensure TB exists (reuse if already present; otherwise add)
626 var newComp = go.GetComponent<TB>();

628 {

630 if (!go.activeInHierarchy)
631 {

633 var def = tmpGO.AddComponent<TextMeshProUGUI>();

635 UnityEditorInternal.ComponentUtility.PasteComponentValues(newComp);
636 UnityEngine.Object.DestroyImmediate(tmpGO);
637 }
638
639 LogReplacement($"Created new component '{newComp.GetType().Name}' on '{newComp.GetPath()}'");
640 if (!newComp)
641 {
642 // catastrophic: restore original situation as best as possible
643 // (re-add TA and blockers)
644 var restoredTA = Undo.AddComponent(go, typeof(TA)) as TA;
645 RestoreGraphicBlockers(go, blockers);
646 UiLog.LogError($"Failed to add target component {typeof(TB).Name} to '{go.GetPath()}'. Rolled back.", go);

649 }

654 _apply?.Invoke(snapshot, newComp);
655
656 // 5) Restore previously removed blockers (Outline/Shadow/etc.)
657 RestoreGraphicBlockers(go, blockers);
658
659 results.Add((snapshot, newComp));
660 }
661
662 EditorSceneManager.MarkSceneDirty(SceneManager.GetActiveScene());
663 Undo.CollapseUndoOperations(Undo.GetCurrentGroup());
664
665 return results;
666#else
667 throw new RoslynUnavailableException();

1054 private static void ProcessNode( List<string> _result, SyntaxNode _node )
1055 {
1056 foreach (var child in _node.ChildNodesAndTokens())
1057 {
1058 if (child.IsNode)
1059 {
1060 var childNode = child.AsNode();
1061
1062 switch (childNode)
1063 {
1064 case LiteralExpressionSyntax literalExpr
1065 when literalExpr.IsKind(SyntaxKind.StringLiteralExpression):
1066 EnsureStringSlot(_result);
1067 _result[_result.Count - 1] += literalExpr.Token.ValueText; // unescaped
1068 break;
1069
1070 case InterpolatedStringTextSyntax interpText:
1071 EnsureStringSlot(_result);
1072 _result[_result.Count - 1] += interpText.TextToken.ValueText; // unescaped
1073 break;
1074
1075 case InterpolationSyntax interp:
1076 // We are currently in a code slot; first, close it by inserting an empty string segment
1077 EnsureStringSlot(_result);
1078
1079 // Now create a fresh code slot for the inner expression
1080 EnsureCodeSlot(_result);
1081
1082 // Recurse only into the inner expression (no braces)
1083 ProcessNode(_result, interp.Expression);
1084
1085 // And ensure a trailing empty string to separate from the following code/text
1086 EnsureStringSlot(_result);
1087 break;
1088 default:
1089 // descend into other nodes
1090 ProcessNode(_result, childNode);
1091 break;
1092 }
1093 }
1094 else
1095 {
1096 // TOKEN: use .Text (no trivia) -> strips comments and whitespace automatically
1097 var token = child.AsToken();
1098 var text = token.Text;
1099 if (!string.IsNullOrEmpty(text))
1100 {
1101 EnsureCodeSlot(_result);
1102 AppendTokenWithMinimalSpace(_result, text);
1103 }
1104 }
1105 }
1106 }

1116 private static void AppendTokenWithMinimalSpace( List<string> _list, string _tokenText )
1117 {
1118 // We are in a code slot by contract
1119 var idx = _list.Count - 1;
1120 var current = _list[idx];
1121
1122 if (NeedsSpace(current, _tokenText))
1123 current += " ";
1124
1125 current += _tokenText;
1126 _list[idx] = current.Trim('"', '$');
1127 }

1136 private static bool NeedsSpace( string _left, string _right )
1137 {
1138 if (string.IsNullOrEmpty(_left) || string.IsNullOrEmpty(_right))
1139 return false;
1140
1141 char lc = _left[_left.Length - 1];
1142 char rc = _right[0];
1143
1144 // add a space only when both sides are "wordy" (identifier-ish)
1145 bool wordLeft = char.IsLetterOrDigit(lc) || lc == '_';
1146 bool wordRight = char.IsLetterOrDigit(rc) || rc == '_';
1147
1148 return wordLeft && wordRight;
1149 }
