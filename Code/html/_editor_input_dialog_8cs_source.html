26 private void OnGUI()
27 {
28 // Check if Esc/Return have been pressed
29 var e = Event.current;
30 if (e.type == EventType.KeyDown)
31 {
32 switch (e.keyCode)
33 {
34 // Escape pressed
35 case KeyCode.Escape:
36 m_shouldClose = true;
37 e.Use();
38 break;
39
40 // Enter pressed
41 case KeyCode.Return:
42 case KeyCode.KeypadEnter:
43 m_onOKButton?.Invoke();
44 m_shouldClose = true;
45 e.Use();
46 break;
47 }
48 }
49
50 if (m_shouldClose)
51 Close();
52
53 // Draw our control
54 var rect = EditorGUILayout.BeginVertical();
55
56 EditorGUILayout.Space(12);
57 EditorGUILayout.LabelField(m_description);
58
59 EditorGUILayout.Space(8);
60 GUI.SetNextControlName("inText");
61 ShowContentField();
62 GUI.FocusControl("inText"); // Focus text field
63 EditorGUILayout.Space(12);
64
65 m_additionalContent?.Invoke(this);
66
67 // Draw OK / Cancel buttons
68 var r = EditorGUILayout.GetControlRect();
69 r.width /= 2;
70 if (GUI.Button(r, m_okButtonText))
71 Ok();
72
73 r.x += r.width;
74 if (GUI.Button(r, m_cancelButtonText))
75 Cancel();
76
77 EditorGUILayout.Space(8);
78 EditorGUILayout.EndVertical();
79
80 // Force change size of the window
81 if (rect.width != 0 && minSize != rect.size)
82 {
83 minSize = maxSize = rect.size;
84 }
85
86 // Set dialog position next to mouse position
87 if (!m_positionInitialized && e.type == EventType.Layout)
88 {
89 m_positionInitialized = true;
90
91 // Move window to a new position. Make sure we're inside visible window
92 var mousePos = GUIUtility.GUIToScreenPoint(Event.current.mousePosition);
93 mousePos.x += 32;
94 if (mousePos.x + position.width > m_maxScreenPos.x)
95 mousePos.x -= position.width + 64; // Display on left side of mouse
96 if (mousePos.y + position.height > m_maxScreenPos.y)
97 mousePos.y = m_maxScreenPos.y - position.height;
98
99 position = new Rect(mousePos.x, mousePos.y, position.width, position.height);
100
101 // Focus current window
102 Focus();
103 }
104 }
