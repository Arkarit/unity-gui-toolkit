141 public void Record()
142 {
143 if ((_support & EStatePropertySupport.SizePosition) != 0)
144 {
145 m_sizeDelta = m_rectTransform.sizeDelta;
146 m_position = m_rectTransform.anchoredPosition;
147 }
148 if ((_support & EStatePropertySupport.Active) != 0)
149 {
150 m_active = m_gameObject.activeSelf;
151 }
152 if ((_support & EStatePropertySupport.Rotation) != 0)
153 {
154 // Aaaand... another Unity frickelation.
155 // It is not possible to get the correct value of the rotation you have entered in the "Transform" editor panel during runtime.
156 // The Unity team in all of their wisdom chose not to populate this value, but instead for eulerAngles etc. only to return values
157 // between 0..180 degrees for positive rotation and 360..180 for negative rotation.
158 // This means, it is not possible to e.g. let an object rotate twice (0..720) since you will always get 0 instead of 720 from eulerAngles.
159 // So we poke around in the serialized Transform via reflection. It contains a serialized property "m_LocalEulerAnglesHint", which holds the real
160 // values the user entered. Of course, this is not available during runtime.
161 SerializedObject serializedTransform = new SerializedObject(m_rectTransform);
162 SerializedProperty localEulerAnglesHint = serializedTransform.FindProperty("m_LocalEulerAnglesHint");
163 m_rotation = localEulerAnglesHint.vector3Value;
164 }
165 if ((_support & EStatePropertySupport.Scale) != 0)
166 {
167 m_scale = m_rectTransform.localScale;
168 }
169 if ((_support & EStatePropertySupport.PreferredWidth) != 0 && m_layoutElement != null)
170 {
171 m_preferredWidth = m_layoutElement.preferredWidth;
172 }
173 if ((_support & EStatePropertySupport.PreferredHeight) != 0 && m_layoutElement != null)
174 {
175 m_preferredHeight = m_layoutElement.preferredHeight;
176 }
177 if ((_support & EStatePropertySupport.Alpha) != 0 && m_canvasGroup != null)
178 {
179 m_alpha = m_canvasGroup.alpha;
180 }
181 if ((_support & EStatePropertySupport.Interactable) != 0 && m_canvasGroup != null)

227 public void Apply( UiState _from, float _normalizedValue )
228 {
229 if (m_rectTransform == null)
230 return;
231
232 if ((_support & EStatePropertySupport.SizePosition) != 0)
233 {
234 m_rectTransform.sizeDelta = Vector2.LerpUnclamped(_from.m_sizeDelta, m_sizeDelta, _normalizedValue);
235 m_rectTransform.anchoredPosition = Vector2.LerpUnclamped(_from.m_position, m_position, _normalizedValue);
236 }
237 if ((_support & EStatePropertySupport.Active) != 0)
238 {
239 m_gameObject.SetActive(LerpBool(_from.m_active, m_active, _normalizedValue));
240 }
241 if ((_support & EStatePropertySupport.Rotation) != 0)
242 {
243 Vector3 rot = Vector3.LerpUnclamped(_from.m_rotation, m_rotation, _normalizedValue);
244 RotateRectTransform(rot);
245 }
246 if ((_support & EStatePropertySupport.Scale) != 0)
247 {
248 m_rectTransform.localScale = Vector3.LerpUnclamped(_from.m_scale, m_scale, _normalizedValue);
249 }
250 if ((_support & EStatePropertySupport.PreferredWidth) != 0 && m_layoutElement != null)
251 {
252 float val = Mathf.LerpUnclamped(_from.m_preferredWidth, m_preferredWidth, _normalizedValue);
253 m_layoutElement.preferredWidth = val;
254 }
255 if ((_support & EStatePropertySupport.PreferredHeight) != 0 && m_layoutElement != null)
256 {
257 float val = Mathf.LerpUnclamped(_from.m_preferredHeight, m_preferredHeight, _normalizedValue);
258 m_layoutElement.preferredHeight = val;
259 }
260 if ((_support & EStatePropertySupport.Alpha) != 0 && m_canvasGroup != null)
261 {
262 float val = Mathf.Lerp(_from.m_alpha, m_alpha, _normalizedValue);

278 private void RotateRectTransform( Vector3 _rotation )
279 {
280#if UNITY_EDITOR
281 SerializedObject serializedTransform = new SerializedObject(m_rectTransform);
282 SerializedProperty localEulerAnglesHint = serializedTransform.FindProperty("m_LocalEulerAnglesHint");
283 Vector3 rotSerialized = localEulerAnglesHint.vector3Value;
284 rotSerialized = _rotation;
285 localEulerAnglesHint.vector3Value = rotSerialized;
286 serializedTransform.ApplyModifiedProperties();
287#endif
288 Vector3 rot = m_rectTransform.eulerAngles;
289 rot = _rotation;
290 m_rectTransform.localRotation = Quaternion.Euler(rot);
