84 public static string EscapeArguments(params string[] args)
85 {
86 StringBuilder arguments = new StringBuilder();
87 Regex invalidChar = new Regex("[\x00\x0a\x0d]"); // these can not be escaped
88 Regex needsQuotes = new Regex(@"\s|"""); // contains whitespace or two quote characters
89 Regex escapeQuote = new Regex(@"(\\*)(""|$)"); // one or more '\' followed with a quote or end of string
90 for (int carg = 0; args != null && carg < args.Length; carg++)
91 {
92 if (args[carg] == null)
93 {
94 throw new ArgumentNullException("args[" + carg + "]");
95 }
96
97 if (invalidChar.IsMatch(args[carg]))
98 {
99 throw new ArgumentOutOfRangeException("args[" + carg + "]");
100 }
101
102 if (args[carg] == String.Empty)
103 {
104 arguments.Append("\"\"");
105 }
106 else if (!needsQuotes.IsMatch(args[carg]))
107 {
108 arguments.Append(args[carg]);
109 }
110 else
111 {
112 arguments.Append('"');
113 arguments.Append(escapeQuote.Replace(args[carg], m =>
114 m.Groups[1].Value + m.Groups[1].Value +
115 (m.Groups[2].Value == "\"" ? "\\\"" : "")
116 ));
117 arguments.Append('"');
118 }
119
120 if (carg + 1 < args.Length)
121 arguments.Append(' ');
122 }
123
124 return arguments.ToString();
