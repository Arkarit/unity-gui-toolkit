257 public void LoadResources()
258 {
259 // cancel any previous run

286 protected T GetAsset<T>( int _index ) where T : Object
287 {
288 if (_index < 0 || _index >= m_loadedAssets.Count)
289 throw new IndexOutOfRangeException($"Asset index {_index} out of range (count {m_loadedAssets.Count}).");
290
291 var obj = m_loadedAssets[_index];
292 if (obj is T t)
293 return t;
294
295 throw new InvalidCastException($"Loaded asset at index {_index} is {obj?.GetType().Name ?? "<null>"}, not {typeof(T).Name}.");
296 }
297
298 protected bool TryGetAsset<T>( int _index, out T _asset ) where T : Object
299 {
300 _asset = null;
301 if (_index < 0 || _index >= m_loadedAssets.Count)
302 return false;
303
304 _asset = m_loadedAssets[_index] as T;
305 return _asset != null;
306 }

308 private async Task LoadAllResourcesAsync( List<CanonicalAssetKey> keys, CancellationToken ct )
309 {
310 try
311 {
312 m_loadedAssets.Clear();
313
314 for (int i = 0; i < keys.Count; i++)
315 {
316 ct.ThrowIfCancellationRequested();
317
318 var key = keys[i];
319
320 // Let provider logic decide (addr:/res:/etc). We request Object handle:
321 // - For CanonicalAssetKey, providers ignore generic T for normalization and honor key.Type.
322 var provider = AssetManager.GetAssetProviderOrThrow(key);
323 var handle = await provider.LoadAssetAsync<Object>(key, ct);
324
325 // track handle for release
326 m_handles.Add(handle);
327
328 // cache the loaded asset (order matches 'resources')
329 m_loadedAssets.Add(handle.Asset);
330 }
331
332 ct.ThrowIfCancellationRequested();
333
334 // everything loaded
335 OnAssetsLoaded();
336 }
337 catch (OperationCanceledException)
338 {
339 // normal on re-init/destroy; swallow
340 }
341 catch (Exception ex)
342 {
343 OnAssetLoadFailed(ex);
344 UiLog.LogError($"[{GetType().Name}] Failed to load resources\n{ex}");
345 }
346 }
347
348 private void ReleaseAllResourceHandles()
349 {
350 if (m_handles.Count > 0)
351 {
352 for (int i = m_handles.Count - 1; i >= 0; i--)
353 try { m_handles[i]?.Release(); } catch { /* swallow */ }
354
355 m_handles.Clear();
356 }
357
358 m_loadedAssets.Clear();
359 }
