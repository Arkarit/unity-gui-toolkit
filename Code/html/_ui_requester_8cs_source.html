327 protected override void EvaluateOptions( UiRequesterBase.Options _options )
328 {
329 base.EvaluateOptions(_options);
330 var options = (Options)_options;
331
332 bool hasText = !string.IsNullOrEmpty(options.Text);
333 bool hasInput =
334 !string.IsNullOrEmpty(options.PlaceholderText)
335 || !string.IsNullOrEmpty(options.InputText);
336 bool hasDateTime = options.DateTimeOptions != null;
337
338 m_textContainer.gameObject.SetActive(hasText);
339 if (hasText)
340 m_text.text = options.Text;
341
342 m_inputFieldContainer.gameObject.SetActive(hasInput);
343
344 if (hasInput)
345 {
346 if (options.InputText != null)
347 {
348 m_inputField.text = options.InputText;
349 }
350 if (options.PlaceholderText != null)
351 {

353 if (placeholderText != null)
354 placeholderText.text = options.PlaceholderText;
355 }
356 UiMain.Instance.SetFocus(m_inputField);
357 }
358
359 m_dateTimePanel.gameObject.SetActive(hasDateTime);
360 if (hasDateTime)
361 m_dateTimePanel.SetOptions(options.DateTimeOptions);
362 }
