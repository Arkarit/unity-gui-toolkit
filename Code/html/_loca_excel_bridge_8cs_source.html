133 public void CollectData()
134 {
135#if UITK_USE_ROSLYN
136 m_processedLoca = null;
137
138 string xlsxPath = string.Empty;
139 switch (m_sourceType)
140 {
141 case SourceType.Local:
142 if (m_excelPath == null || string.IsNullOrEmpty(m_excelPath.Path))
143 {
144 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel path is not set.");
145 return;
146 }
147
148 xlsxPath = m_excelPath.Path;
149 break;
150
151 case SourceType.GoogleDocs:
152 xlsxPath = ProcessGoogleDocsUrl();
153 if (string.IsNullOrEmpty(xlsxPath))
154 return;
155
156 break;
157
158 default:
159 throw new ArgumentOutOfRangeException();
160 }
161
162 if (!File.Exists(xlsxPath))
163 {
164 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel file not found: {xlsxPath}");
165 return;
166 }
167
168 Encoding.RegisterProvider(CodePagesEncodingProvider.Instance);
169
170 using var fs = File.Open(xlsxPath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite);
171 using var reader = ExcelReaderFactory.CreateReader(fs);
172 DataSet ds = reader.AsDataSet();
173 int tablesCount = ds.Tables.Count;
174 if (tablesCount == 0)
175 {
176 UiLog.LogError($"{nameof(LocaExcelBridge)}: No worksheets found in {xlsxPath}");
177 return;
178 }
179
180 var sheet = ds.Tables[0];
181 int colCount = sheet.Columns.Count;
182
183 for (int i = 1; i < tablesCount; i++)
184 {
185 var xlsxColCount = ds.Tables[i].Columns.Count;
186 if (xlsxColCount < colCount)
187 {
188 UiLog.LogError($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too small for defined columns ({colCount})");
189 return;
190 }
191
192 if (xlsxColCount > colCount)
193 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too large for defined columns ({colCount}). Ignored.");
194 }
195
196 if (sheet.Rows.Count <= m_startRow)
197 {
198 UiLog.LogError($"{nameof(LocaExcelBridge)}: Worksheet has no data rows after start row {m_startRow}.");
199 return;
200 }
201
202
203 // Infer column config if not matching column count
204 if (m_columnDescriptions == null || m_columnDescriptions.Count != colCount)
205 {
206 m_columnDescriptions = new List<InColumnDescription>(colCount);
207 for (int c = 0; c < colCount; c++)
208 {
209 m_columnDescriptions.Add(new InColumnDescription
210 {
211 ColumnType = (c == 0) ? EInColumnType.Key : EInColumnType.LanguageTranslation,
212 KeyPrefix = string.Empty,
213 KeyPostfix = string.Empty,
214 LanguageId = string.Empty,
215 PluralForm = -1
216 });
217 }
218 }
219
220 int keyCol = -1;
221 InColumnDescription keyColDesc = null;
222 var langColumns = new List<(int col, string lang, InColumnDescription desc)>();
223
224 for (int c = 0; c < colCount; c++)
225 {
226 var desc = m_columnDescriptions[c];
227 if (desc == null)
228 continue;
229
230 switch (desc.ColumnType)
231 {
232 case EInColumnType.Key:
233 keyCol = c;
234 keyColDesc = desc;
235 break;
236
237 case EInColumnType.LanguageTranslation:
238 string langId = NormalizeLang(desc.LanguageId);
239 if (string.IsNullOrEmpty(langId))
240 {
241 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Empty LanguageId at column {c}, skipping.");
242 continue;
243 }
244 langColumns.Add((c, langId, desc));
245 break;
246 }
247 }
248
249 if (keyCol < 0)
250 {
251 UiLog.LogError($"{nameof(LocaExcelBridge)}: No key column defined or inferred.");
252 return;
253 }
254
255 var byLangAndKey = new Dictionary<(string lang, string key), ProcessedLocaEntry>(1024, StringTupleComparer.Ordinal);
256
257 for (int i = 0; i < tablesCount; i++)
258 {
259 sheet = ds.Tables[i];
260 for (int r = m_startRow; r < sheet.Rows.Count; r++)
261 {
262 string baseKey = sheet.Rows[r][keyCol]?.ToString()?.Trim();
263 if (string.IsNullOrEmpty(baseKey))
264 continue;
265
266 string baseEffectiveKey = ApplyKeyAffixes(baseKey, keyColDesc);
267
268 foreach (var lc in langColumns)
269 {
270 string lang = lc.lang;
271 string cell = sheet.Rows[r][lc.col]?.ToString();
272 cell = cell != null ? cell.Trim() : string.Empty;
273
274 int plural = lc.desc?.PluralForm ?? -1;
275
276 // skip empty cells
277 if (string.IsNullOrEmpty(cell))
278 continue;
279
280 string effectiveKey = ApplyKeyAffixes(baseEffectiveKey, lc.desc);
281
282 var k = (lang, effectiveKey);
283 if (!byLangAndKey.TryGetValue(k, out var entry))
284 {
285 entry = new ProcessedLocaEntry { LanguageId = lang, Key = effectiveKey };
286 byLangAndKey[k] = entry;
287 }
288
289 if (plural < 0)
290 {
291 entry.Text = cell;
292 }
293 else
294 {
295 entry.Forms ??= new string[6];
296 entry.Forms[plural] = cell;
297 }
298 }
299 }
300 }
301
302 var pruned = byLangAndKey.Values
303 .Where(e =>
304 !string.IsNullOrEmpty(e.Text) ||
305 (e.Forms != null && e.Forms.Any(s => !string.IsNullOrEmpty(s)))
306 )
307 .ToList();
308
309 m_processedLoca = new ProcessedLoca(m_group, pruned);
310
311 EditorUtility.SetDirty(this);
312 AssetDatabase.Refresh();
313#else
314 throw new RoslynUnavailableException();
315#endif

320 private string ProcessGoogleDocsUrl()
321 {
322 string tempFile = Path.GetTempFileName();
323 string path = NormalizeXlsxPath(m_googleUrl);
324 if (string.IsNullOrWhiteSpace(path))
325 return null;
326
327 using (UnityWebRequest www = UnityWebRequest.Get(path))
328 {
329 var op = www.SendWebRequest();
330 var sw = Stopwatch.StartNew();
331 const int timeoutMs = 60000;
332
333 while (!op.isDone)
334 {
335 if (sw.ElapsedMilliseconds > timeoutMs)
336 {
337 www.Abort();
338 UiLog.LogError($"{nameof(LocaExcelBridge)}: Download timeout for XLSX:{path}");
339 return null;
340 }
341
342 EditorApplication.QueuePlayerLoopUpdate();
343 Thread.Sleep(10);
344 }
345
346 if (www.result != UnityWebRequest.Result.Success)
347 {
348 UiLog.LogError($"{nameof(LocaExcelBridge)}: Failed to download XLSX: {www.error}");
349 return null;
350 }
351
352 File.WriteAllBytes(tempFile, www.downloadHandler.data);
353 return tempFile;
354 }

357 private static string NormalizeXlsxPath( string _path )
358 {
359 // Empty -> fail
360 if (string.IsNullOrWhiteSpace(_path))
361 {
362 UiLog.LogError("Error: <empty> is not a valid path");
363 return null;
364 }
365
366 // Local or network path -> fail
367 if (!_path.StartsWith("http", StringComparison.OrdinalIgnoreCase))
368 {
369 UiLog.LogError($"Error: '{_path}' is not a valid URL");
370 return null;
371 }
372
373 // Already an export URL with XLSX -> unchanged
374 if (_path.Contains("/export", StringComparison.OrdinalIgnoreCase)
375 && _path.Contains("format=xlsx", StringComparison.OrdinalIgnoreCase))
376 {
377 return _path;
378 }
379
380 // Try to treat it as a normal Google Sheets "edit" URL.
381 const string marker = "/d/";
382 int markerIndex = _path.IndexOf(marker, StringComparison.Ordinal);
383 if (markerIndex < 0)
384 {
385 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Google Sheets URL does not contain '/d/': {_path}");
386 return null;
387 }
388
389 int idStart = markerIndex + marker.Length;
390 int idEnd = _path.IndexOfAny(new[] { '/', '?', '#' }, idStart);
391
392 string docId;
393 if (idEnd >= 0)
394 docId = _path.Substring(idStart, idEnd - idStart);
395 else
396 docId = _path.Substring(idStart);
397
398 if (string.IsNullOrEmpty(docId))
399 {
400 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Could not extract document id from Google Sheets URL: {_path}");
401 return null;
402 }
403
404 string exportUrl = $"https://docs.google.com/spreadsheets/d/{docId}/export?format=xlsx&id={docId}";
405 return exportUrl;
406 }
