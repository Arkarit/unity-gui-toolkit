270 public void CollectData()
271 {
272#if UITK_USE_ROSLYN
273 m_processedLoca = null;
274
275 string xlsxPath = string.Empty;
276 switch (m_sourceType)
277 {
278 case SourceType.Local:
279 if (m_excelPath == null || string.IsNullOrEmpty(m_excelPath.Path))
280 {
281 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel path is not set.");
282 return;
283 }
284
285 xlsxPath = m_excelPath.Path;
286 break;
287
288 case SourceType.GoogleDocs:
289 xlsxPath = ProcessGoogleDocsUrl();
290 if (string.IsNullOrEmpty(xlsxPath))
291 return;
292
293 break;
294
295 default:
296 throw new ArgumentOutOfRangeException();
297 }
298
299 if (!File.Exists(xlsxPath))
300 {
301 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel file not found: {xlsxPath}");
302 return;
303 }
304
305 Encoding.RegisterProvider(CodePagesEncodingProvider.Instance);
306
307 using var fs = File.Open(xlsxPath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite);
308 using var reader = ExcelReaderFactory.CreateReader(fs);
309 DataSet ds = reader.AsDataSet();
310 int tablesCount = ds.Tables.Count;
311 if (tablesCount == 0)
312 {
313 UiLog.LogError($"{nameof(LocaExcelBridge)}: No worksheets found in {xlsxPath}");
314 return;
315 }
316
317 var sheet = ds.Tables[0];
318 int colCount = sheet.Columns.Count;
319
320 for (int i = 1; i < tablesCount; i++)
321 {
322 var xlsxColCount = ds.Tables[i].Columns.Count;
323 if (xlsxColCount < colCount)
324 {
325 UiLog.LogError($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too small for defined columns ({colCount})");
326 return;
327 }
328
329 if (xlsxColCount > colCount)
330 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too large for defined columns ({colCount}). Ignored.");
331 }
332
333 if (sheet.Rows.Count <= m_startRow)
334 {
335 UiLog.LogError($"{nameof(LocaExcelBridge)}: Worksheet has no data rows after start row {m_startRow}.");
336 return;
337 }
338
339
340 // Infer column config if not matching column count
341 if (m_columnDescriptions == null || m_columnDescriptions.Count != colCount)
342 {
343 m_columnDescriptions = new List<InColumnDescription>(colCount);
344 for (int c = 0; c < colCount; c++)
345 {
346 m_columnDescriptions.Add(new InColumnDescription
347 {
348 ColumnType = (c == 0) ? EInColumnType.Key : EInColumnType.LanguageTranslation,
349 KeyPrefix = string.Empty,
350 KeyPostfix = string.Empty,
351 LanguageId = string.Empty,
352 PluralForm = -1
353 });
354 }
355 }
356
357 int keyCol = -1;
358 InColumnDescription keyColDesc = null;
359 var langColumns = new List<(int col, string lang, InColumnDescription desc)>();
360
361 for (int c = 0; c < colCount; c++)
362 {
363 var desc = m_columnDescriptions[c];
364 if (desc == null)
365 continue;
366
367 switch (desc.ColumnType)
368 {
369 case EInColumnType.Key:
370 keyCol = c;
371 keyColDesc = desc;
372 break;
373
374 case EInColumnType.LanguageTranslation:
375 string langId = NormalizeLang(desc.LanguageId);
376 if (string.IsNullOrEmpty(langId))
377 {
378 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Empty LanguageId at column {c}, skipping.");
379 continue;
380 }
381 langColumns.Add((c, langId, desc));
382 break;
383 }
384 }
385
386 if (keyCol < 0)
387 {
388 UiLog.LogError($"{nameof(LocaExcelBridge)}: No key column defined or inferred.");
389 return;
390 }
391
392 var byLangAndKey = new Dictionary<(string lang, string key), ProcessedLocaEntry>(1024, StringTupleComparer.Ordinal);
393
394 for (int i = 0; i < tablesCount; i++)
395 {
396 sheet = ds.Tables[i];
397 for (int r = m_startRow; r < sheet.Rows.Count; r++)
398 {
399 string baseKey = sheet.Rows[r][keyCol]?.ToString()?.Trim();
400 if (string.IsNullOrEmpty(baseKey))
401 continue;
402
403 string baseEffectiveKey = ApplyKeyAffixes(baseKey, keyColDesc);
404
405 foreach (var lc in langColumns)
406 {
407 string lang = lc.lang;
408 string cell = sheet.Rows[r][lc.col]?.ToString();
409 cell = cell != null ? cell.Trim() : string.Empty;
410
411 int plural = lc.desc?.PluralForm ?? -1;
412
413 // skip empty cells
414 if (string.IsNullOrEmpty(cell))
415 continue;
416
417 string effectiveKey = ApplyKeyAffixes(baseEffectiveKey, lc.desc);
418
419 var k = (lang, effectiveKey);
420 if (!byLangAndKey.TryGetValue(k, out var entry))
421 {
422 entry = new ProcessedLocaEntry { LanguageId = lang, Key = effectiveKey };
423 byLangAndKey[k] = entry;
424 }
425
426 if (plural < 0)
427 {
428 entry.Text = cell;
429 }
430 else
431 {
432 entry.Forms ??= new string[6];
433 entry.Forms[plural] = cell;
434 }
435 }
436 }
437 }
438
439 var pruned = byLangAndKey.Values
440 .Where(e =>
441 !string.IsNullOrEmpty(e.Text) ||
442 (e.Forms != null && e.Forms.Any(s => !string.IsNullOrEmpty(s)))
443 )
444 .ToList();
445
446 m_processedLoca = new ProcessedLoca(m_group, pruned);
447
448 EditorUtility.SetDirty(this);
449 AssetDatabase.Refresh();
450#else
451 throw new RoslynUnavailableException();
452#endif
453 }

457 private string ProcessGoogleDocsUrl()
458 {
459 string tempFile = Path.GetTempFileName();
460 string path = NormalizeXlsxPath(m_googleUrl);
461 if (string.IsNullOrWhiteSpace(path))
462 return null;
463
464 string bearerToken = null;
465 if (m_useGoogleAuth && !string.IsNullOrEmpty(m_serviceAccountJsonPath))
466 {
467 bearerToken = GoogleServiceAccountAuth.GetAccessToken(m_serviceAccountJsonPath);
468 if (bearerToken == null)
469 {
470 UiLog.LogError($"{nameof(LocaExcelBridge)}: Failed to obtain Google auth token");
471 return null;
472 }
473 }
474
475 using (UnityWebRequest www = UnityWebRequest.Get(path))
476 {
477 if (bearerToken != null)
478 www.SetRequestHeader("Authorization", $"Bearer {bearerToken}");
479
480 var op = www.SendWebRequest();
481 var sw = Stopwatch.StartNew();
482 const int timeoutMs = 60000;
483
484 while (!op.isDone)
485 {
486 if (sw.ElapsedMilliseconds > timeoutMs)
487 {
488 www.Abort();
489 UiLog.LogError($"{nameof(LocaExcelBridge)}: Download timeout for XLSX:{path}");
490 return null;
491 }
492
493 EditorApplication.QueuePlayerLoopUpdate();
494 Thread.Sleep(10);
495 }
496
497 if (www.result != UnityWebRequest.Result.Success)
498 {
499 UiLog.LogError($"{nameof(LocaExcelBridge)}: Failed to download XLSX: {www.error}");
500 return null;
501 }
502
503 File.WriteAllBytes(tempFile, www.downloadHandler.data);
504 return tempFile;
505 }

508 private static string NormalizeXlsxPath( string _path )
509 {
510 // Empty -> fail
511 if (string.IsNullOrWhiteSpace(_path))
512 {
513 UiLog.LogError("Error: <empty> is not a valid path");
514 return null;
515 }
516
517 // Local or network path -> fail
518 if (!_path.StartsWith("http", StringComparison.OrdinalIgnoreCase))
519 {
520 UiLog.LogError($"Error: '{_path}' is not a valid URL");
521 return null;
522 }
523
524 // Already an export URL with XLSX -> unchanged
525 if (_path.Contains("/export", StringComparison.OrdinalIgnoreCase)
526 && _path.Contains("format=xlsx", StringComparison.OrdinalIgnoreCase))
527 {
528 return _path;
529 }
530
531 // Try to treat it as a normal Google Sheets "edit" URL.
532 const string marker = "/d/";
533 int markerIndex = _path.IndexOf(marker, StringComparison.Ordinal);
534 if (markerIndex < 0)
535 {
536 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Google Sheets URL does not contain '/d/': {_path}");
537 return null;
538 }
539
540 int idStart = markerIndex + marker.Length;
541 int idEnd = _path.IndexOfAny(new[] { '/', '?', '#' }, idStart);
542
543 string docId;
544 if (idEnd >= 0)
545 docId = _path.Substring(idStart, idEnd - idStart);
546 else
547 docId = _path.Substring(idStart);
548
549 if (string.IsNullOrEmpty(docId))
550 {
551 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Could not extract document id from Google Sheets URL: {_path}");
552 return null;
553 }
554
555 string exportUrl = $"https://docs.google.com/spreadsheets/d/{docId}/export?format=xlsx&id={docId}";
556 return exportUrl;
557 }
