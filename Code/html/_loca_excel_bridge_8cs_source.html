198 public void CollectData()
199 {
200#if UITK_USE_ROSLYN
201 m_processedLoca = null;
202
203 string xlsxPath = string.Empty;
204 switch (m_sourceType)
205 {
206 case SourceType.Local:
207 if (m_excelPath == null || string.IsNullOrEmpty(m_excelPath.Path))
208 {
209 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel path is not set.");
210 return;
211 }
212
213 xlsxPath = m_excelPath.Path;
214 break;
215
216 case SourceType.GoogleDocs:
217 xlsxPath = ProcessGoogleDocsUrl();
218 if (string.IsNullOrEmpty(xlsxPath))
219 return;
220
221 break;
222
223 default:
224 throw new ArgumentOutOfRangeException();
225 }
226
227 if (!File.Exists(xlsxPath))
228 {
229 UiLog.LogError($"{nameof(LocaExcelBridge)}: Excel file not found: {xlsxPath}");
230 return;
231 }
232
233 Encoding.RegisterProvider(CodePagesEncodingProvider.Instance);
234
235 using var fs = File.Open(xlsxPath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite);
236 using var reader = ExcelReaderFactory.CreateReader(fs);
237 DataSet ds = reader.AsDataSet();
238 int tablesCount = ds.Tables.Count;
239 if (tablesCount == 0)
240 {
241 UiLog.LogError($"{nameof(LocaExcelBridge)}: No worksheets found in {xlsxPath}");
242 return;
243 }
244
245 var sheet = ds.Tables[0];
246 int colCount = sheet.Columns.Count;
247
248 for (int i = 1; i < tablesCount; i++)
249 {
250 var xlsxColCount = ds.Tables[i].Columns.Count;
251 if (xlsxColCount < colCount)
252 {
253 UiLog.LogError($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too small for defined columns ({colCount})");
254 return;
255 }
256
257 if (xlsxColCount > colCount)
258 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Column count ({ds.Tables[i].Columns.Count}) too large for defined columns ({colCount}). Ignored.");
259 }
260
261 if (sheet.Rows.Count <= m_startRow)
262 {
263 UiLog.LogError($"{nameof(LocaExcelBridge)}: Worksheet has no data rows after start row {m_startRow}.");
264 return;
265 }
266
267
268 // Infer column config if not matching column count
269 if (m_columnDescriptions == null || m_columnDescriptions.Count != colCount)
270 {
271 m_columnDescriptions = new List<InColumnDescription>(colCount);
272 for (int c = 0; c < colCount; c++)
273 {
274 m_columnDescriptions.Add(new InColumnDescription
275 {
276 ColumnType = (c == 0) ? EInColumnType.Key : EInColumnType.LanguageTranslation,
277 KeyPrefix = string.Empty,
278 KeyPostfix = string.Empty,
279 LanguageId = string.Empty,
280 PluralForm = -1
281 });
282 }
283 }
284
285 int keyCol = -1;
286 InColumnDescription keyColDesc = null;
287 var langColumns = new List<(int col, string lang, InColumnDescription desc)>();
288
289 for (int c = 0; c < colCount; c++)
290 {
291 var desc = m_columnDescriptions[c];
292 if (desc == null)
293 continue;
294
295 switch (desc.ColumnType)
296 {
297 case EInColumnType.Key:
298 keyCol = c;
299 keyColDesc = desc;
300 break;
301
302 case EInColumnType.LanguageTranslation:
303 string langId = NormalizeLang(desc.LanguageId);
304 if (string.IsNullOrEmpty(langId))
305 {
306 UiLog.LogWarning($"{nameof(LocaExcelBridge)}: Empty LanguageId at column {c}, skipping.");
307 continue;
308 }
309 langColumns.Add((c, langId, desc));
310 break;
311 }
312 }
313
314 if (keyCol < 0)
315 {
316 UiLog.LogError($"{nameof(LocaExcelBridge)}: No key column defined or inferred.");
317 return;
318 }
319
320 var byLangAndKey = new Dictionary<(string lang, string key), ProcessedLocaEntry>(1024, StringTupleComparer.Ordinal);
321
322 for (int i = 0; i < tablesCount; i++)
323 {
324 sheet = ds.Tables[i];
325 for (int r = m_startRow; r < sheet.Rows.Count; r++)
326 {
327 string baseKey = sheet.Rows[r][keyCol]?.ToString()?.Trim();
328 if (string.IsNullOrEmpty(baseKey))
329 continue;
330
331 string baseEffectiveKey = ApplyKeyAffixes(baseKey, keyColDesc);
332
333 foreach (var lc in langColumns)
334 {
335 string lang = lc.lang;
336 string cell = sheet.Rows[r][lc.col]?.ToString();
337 cell = cell != null ? cell.Trim() : string.Empty;
338
339 int plural = lc.desc?.PluralForm ?? -1;
340
341 // skip empty cells
342 if (string.IsNullOrEmpty(cell))
343 continue;
344
345 string effectiveKey = ApplyKeyAffixes(baseEffectiveKey, lc.desc);
346
347 var k = (lang, effectiveKey);
348 if (!byLangAndKey.TryGetValue(k, out var entry))
349 {
350 entry = new ProcessedLocaEntry { LanguageId = lang, Key = effectiveKey };
351 byLangAndKey[k] = entry;
352 }
353
354 if (plural < 0)
355 {
356 entry.Text = cell;
357 }
358 else
359 {
360 entry.Forms ??= new string[6];
361 entry.Forms[plural] = cell;
362 }
363 }
364 }
365 }
366
367 var pruned = byLangAndKey.Values
368 .Where(e =>
369 !string.IsNullOrEmpty(e.Text) ||
370 (e.Forms != null && e.Forms.Any(s => !string.IsNullOrEmpty(s)))
371 )
372 .ToList();
373
374 m_processedLoca = new ProcessedLoca(m_group, pruned);
375
376 EditorUtility.SetDirty(this);
377 AssetDatabase.Refresh();
378#else
379 throw new RoslynUnavailableException();
380#endif
381 }

385 private string ProcessGoogleDocsUrl()
386 {
387 string tempFile = Path.GetTempFileName();
388 string path = NormalizeXlsxPath(m_googleUrl);
389 if (string.IsNullOrWhiteSpace(path))
390 return null;
391
392 string bearerToken = null;
393 if (m_useGoogleAuth && !string.IsNullOrEmpty(m_serviceAccountJsonPath))
394 {
395 bearerToken = GoogleServiceAccountAuth.GetAccessToken(m_serviceAccountJsonPath);
396 if (bearerToken == null)
397 {
398 UiLog.LogError($"{nameof(LocaExcelBridge)}: Failed to obtain Google auth token");
399 return null;
400 }
401 }
402
403 using (UnityWebRequest www = UnityWebRequest.Get(path))
404 {
405 if (bearerToken != null)
406 www.SetRequestHeader("Authorization", $"Bearer {bearerToken}");
407
408 var op = www.SendWebRequest();
409 var sw = Stopwatch.StartNew();
410 const int timeoutMs = 60000;
411
412 while (!op.isDone)
413 {
414 if (sw.ElapsedMilliseconds > timeoutMs)
415 {
416 www.Abort();
417 UiLog.LogError($"{nameof(LocaExcelBridge)}: Download timeout for XLSX:{path}");
418 return null;
419 }
420
421 EditorApplication.QueuePlayerLoopUpdate();
422 Thread.Sleep(10);
423 }
424
425 if (www.result != UnityWebRequest.Result.Success)
426 {
427 UiLog.LogError($"{nameof(LocaExcelBridge)}: Failed to download XLSX: {www.error}");
428 return null;
429 }
430
431 File.WriteAllBytes(tempFile, www.downloadHandler.data);
432 return tempFile;
433 }

436 private static string NormalizeXlsxPath( string _path )
437 {
438 // Empty -> fail
439 if (string.IsNullOrWhiteSpace(_path))
440 {
441 UiLog.LogError("Error: <empty> is not a valid path");
442 return null;
443 }
444
445 // Local or network path -> fail
446 if (!_path.StartsWith("http", StringComparison.OrdinalIgnoreCase))
447 {
448 UiLog.LogError($"Error: '{_path}' is not a valid URL");
449 return null;
450 }
451
452 // Already an export URL with XLSX -> unchanged
453 if (_path.Contains("/export", StringComparison.OrdinalIgnoreCase)
454 && _path.Contains("format=xlsx", StringComparison.OrdinalIgnoreCase))
455 {
456 return _path;
457 }
458
459 // Try to treat it as a normal Google Sheets "edit" URL.
460 const string marker = "/d/";
461 int markerIndex = _path.IndexOf(marker, StringComparison.Ordinal);
462 if (markerIndex < 0)
463 {
464 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Google Sheets URL does not contain '/d/': {_path}");
465 return null;
466 }
467
468 int idStart = markerIndex + marker.Length;
469 int idEnd = _path.IndexOfAny(new[] { '/', '?', '#' }, idStart);
470
471 string docId;
472 if (idEnd >= 0)
473 docId = _path.Substring(idStart, idEnd - idStart);
474 else
475 docId = _path.Substring(idStart);
476
477 if (string.IsNullOrEmpty(docId))
478 {
479 UiLog.LogError($"Error: {nameof(LocaExcelBridge)}: Could not extract document id from Google Sheets URL: {_path}");
480 return null;
481 }
482
483 string exportUrl = $"https://docs.google.com/spreadsheets/d/{docId}/export?format=xlsx&id={docId}";
484 return exportUrl;
485 }
