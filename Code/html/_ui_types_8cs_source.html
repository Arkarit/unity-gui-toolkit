82 public float AspectRatio => Height == 0 ? 0 : Width / Height;

84 public bool IsLandscape => AspectRatio >= 1;

86 public readonly bool IsInvalid => Height <= 0 || Width <= 0;

99 public int CompareTo( object _obj )
100 {
101 if (_obj is not ScreenResolution other)
102 return 1;
103
104 float a = AspectRatio;
105 float b = other.AspectRatio;
106
107 int aspectCompare = a.CompareTo(b);
108 if (aspectCompare != 0)
109 return aspectCompare;
110
111 // Secondary sort: by total pixel area (Width * Height)
112 float areaA = Width * Height;
113 float areaB = other.Width * other.Height;
114
115 return areaA.CompareTo(areaB);
116 }

120 return $"{Width}:{Height}";
