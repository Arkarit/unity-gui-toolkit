80 }
81 else

83 result.AddedKeys++;
84 result.AddedKeysList.Add(key);
85 }
86
87 merged.Entries.Add(newEntry);
88 }
89
90 // Keys in PO not in POT become obsolete (or dropped)
91 foreach (var existingEntry in _existingPo.Entries)
92 {
93 if (existingEntry.IsObsolete)

95 merged.Entries.Add(CloneEntry(existingEntry));
