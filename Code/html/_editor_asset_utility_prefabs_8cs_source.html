36 public static void SortByPrefabHierarchy(List<GameObject> _gameObjectList)
37 {
38 ValidateList(_gameObjectList);
39 ValidateArguments(_gameObjectList);
40
41 List<GameObject> toDo = new List<GameObject>(_gameObjectList);
42 List<GameObject> done = new List<GameObject>(); // List to hold sorted GameObjects
43
44 List<Transform[]> transformsCache = new();
45 foreach (var gameObject in toDo)
46 transformsCache.Add(gameObject.GetComponentsInChildren<Transform>());
47
48 while (toDo.Count > 0)
49 {
50 // Iterate through the list in reverse order to avoid issues when removing items
51 for (int i = toDo.Count - 1; i >= 0; i--)
52 {
53 GameObject current = toDo[i];
54 var transforms = transformsCache[i];
55
56 if (ContainsUnhandledPrefabs(transforms, current))
57 continue;
58
59 if (PrefabUtility.IsPartOfVariantPrefab(current))
60 {
61 var basePrefab = PrefabUtility.GetCorrespondingObjectFromSource(current);
62 if (toDo.Contains(basePrefab))
63 continue;
64 }
65
66 // Mark this GameObject as done, and remove it from the "to-do" list and also remove transform from transform cache
67 done.Add(current);
68 toDo.RemoveAt(i);
69 transformsCache.RemoveAt(i);
70
71 // This "break" is very counterintuitive, but it is crucial,
72 // because once we successfully mark a GameObject as done (processed),
73 // we need to stop the current iteration and start over with the remaining
74 // GameObjects in the "to-do" list. This ensures that we don't accidentally
75 // skip any GameObjects that depend on the current one, and we can process
76 // them in the correct order. Without this, we would risk processing a
77 // GameObject before its dependencies, causing the hierarchy to be incorrectly sorted.
78 break;
79 }
80 }
81
82 _gameObjectList.Clear();
83 _gameObjectList.AddRange(done);

85 // Helper function to check if a GameObject has unprocessed prefabs in its hierarchy
86 bool ContainsUnhandledPrefabs(Transform[] _transformArray, GameObject _gameObject)
87 {
88 foreach (var t in _transformArray)
89 {
90 GameObject currentDescendant = t.gameObject;
91
92 if (currentDescendant == _gameObject)
93 continue;
94
95 if (!PrefabUtility.IsAnyPrefabInstanceRoot(currentDescendant))
96 continue;
97
98 // Get the original prefab and check if it is still in the "to-do" list
99 var prefab = PrefabUtility.GetCorrespondingObjectFromOriginalSource(currentDescendant);
100 if (toDo.Contains(prefab))
101 return true;
102 }
103
104 return false;
105 }
