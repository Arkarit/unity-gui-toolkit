216 return new CanonicalAssetKey(this, id, obj.GetType());
217#else
218 if (_key is Object obj)
219 throw new InvalidOperationException($"Object keys are not supported. Use a '{Prefix}' path instead.");

269 public Object Load(CanonicalAssetKey _assetKey, CancellationToken _cancellationToken)
270 {
271 Object result = null;
272
273 if (_assetKey.TryGetValue(Prefix, out string resourcePath))
274 {
275 // Components cannot be loaded directly via Resources.
276 // We load a GameObject and (optionally) verify the required component type exists.
277 if (typeof(Component).IsAssignableFrom(_assetKey.Type))
278 {
279 var go = Resources.Load(resourcePath, typeof(GameObject)) as GameObject;
280 if (!go)
281 throw new AssetLoadFailedException(_assetKey);
282
283 var component = go.GetComponent(_assetKey.Type);
284 if (!component)
285 {
286 go.SafeDestroy();
287 throw new AssetLoadFailedException(_assetKey,
288 $"Asset was found, but didn't carry a Component of type '{_assetKey.Type}'");
289 }
290
291 result = go;
292 }
293 else
294 {
295 result = Resources.Load(resourcePath, _assetKey.Type);
296 }
297 }
298
299 _cancellationToken.ThrowIfCancellationRequested();
300
301 if (!result)
302 throw new AssetLoadFailedException(_assetKey);
303
304 return result;
305 }
