157 private static bool Evaluate( string _path, List<string> _strings, ref int _idx, string _expectedKeyword, bool _expectsPlural )
158 {
159 // Peek helpers
160 bool AreTwoTokensLeft( int _idx ) => _idx <= _strings.Count - 2;
161 bool IsOneTokenLeft( int _idx ) => _idx <= _strings.Count - 1;
162
163 bool Error( string _message, int _idx )
164 {
165 UiLog.LogError($"Loca parsing error '{_message}'\n" +
166 $"in '{_path}' near line {_idx}");
167 return false;
168 }
169
170 // Peek keyword + locaKey without consuming on mismatch
171 if (!AreTwoTokensLeft(_idx))
172 return false;
173
174 if (_strings[_idx] == null)
175 return Error("null string", _idx);
176
177 string keyword = _strings[_idx].RemoveWhitespace();
178 string locaKey = _strings[_idx + 1];
179
180 if (!keyword.EndsWith(_expectedKeyword, StringComparison.Ordinal))
181 return false;
182
183 // Now consume the two tokens
184 _idx += 2;
185
186 if (string.IsNullOrEmpty(locaKey))
187 return Error("Syntax error: empty loca key", _idx);
188
189 string locaKeyPlural = null;
190 if (_expectsPlural)
191 {
192 if (!AreTwoTokensLeft(_idx))
193 return Error("Unexpected end of file (plural)", _idx);
194
195 string comma = _strings[_idx++];
196 string pluralKey = _strings[_idx++];
197
198 if (comma.Trim() != ",")
199 return Error("Syntax error: missing ',' in plural", _idx);
200
201 if (string.IsNullOrEmpty(pluralKey))
202 return Error("Syntax error: empty plural key", _idx);
203
204 locaKeyPlural = pluralKey;
205 }
206
207 string groupKey = null;
208 if (AreTwoTokensLeft(_idx))
209 {
210 var s = _strings[_idx].Trim();
211 if (s.StartsWith(",") && !s.Contains(")"))
212 {
213 _idx++; // consume comma
214 if (!IsOneTokenLeft(_idx))
215 return Error("Unexpected end after group comma", _idx);
216
217 groupKey = _strings[_idx++];
218 if (groupKey == string.Empty)
219 groupKey = null;
220 }
221 }
222
223 LocaManager.Instance.EdAddKey(locaKey, locaKeyPlural, groupKey);
224
225 // Found and added Key
226 return true;
227 }
