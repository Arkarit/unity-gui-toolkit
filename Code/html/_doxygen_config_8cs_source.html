78 private string GetGitVersion()
79 {
80 var currentWorkingDir = Environment.CurrentDirectory;
81 var workingFolder = EditorFileUtility.FindGitFolderInParent(currentWorkingDir);
82 if (workingFolder == null)
83 {
84 string s = "Error: No .git folder found";
85 UiLog.LogError(s);
86 return s;
87 }
88
89 workingFolder += "/";
90
91 string result = string.Empty;
92 int processResult = 0;
93
94 try
95 {
96 // TODO Mac and Linux version
97 processResult = EditorProcessHelper.Run(null, "git.exe", workingFolder, s => result = s, "describe", "--tags", "--abbrev=0");
98 }
99 catch (Exception e)
100 {
101 UiLog.LogError($"Exception when starting Git:{e.Message}");
102 return "Git Exception";
103 }
104
105 if (processResult != 0)
106 {
107 string s = $"Error: Git exited with code {processResult}";
108 UiLog.LogError(s);
109 return s;
110 }
111
112 return result;
113 }
