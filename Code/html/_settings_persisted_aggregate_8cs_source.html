14 public void Apply( Dictionary<string, PlayerSetting> _settings )
15 {
16 foreach (var kv in _settings)
17 {
18 var setting = kv.Value;
19 if (!setting.Options.IsSaveable)
20 continue;
21
22 var key = kv.Key;
23 var value = setting.Value;
24
25 if (string.IsNullOrEmpty(key))
26 {
27 UiLog.LogError("Key is empty!");
28 continue;
29 }
30
31 switch (value)
32 {
33 case int i:
34 m_state.ints[key] = i;
35 break;
36 case float f:
37 m_state.floats[key] = f;
38 break;
39 case string s:
40 m_state.strings[key] = s;
41 break;
42 case bool b:
43 m_state.bools[key] = b;
44 break;
45 case System.Enum e:
46 m_state.ints[key] = System.Convert.ToInt32(e);
47 break;
48 case KeyBinding kb:
49 m_state.ints[key] = kb.Encoded;
50 break;
51 default:
52 UiLog.LogError($"Can not apply value for key '{key}'; unsupported type '{value?.GetType()}'");
53 break;
54 }
