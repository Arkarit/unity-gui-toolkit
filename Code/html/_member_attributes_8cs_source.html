228 private Object FindContextObject( SerializedProperty _property )
229 {
230 var attr = (MandatoryExternalAttribute)attribute;
231 if (string.IsNullOrEmpty(attr.ContextTypeName))
232 return null;
233
234 var target = _property.serializedObject.targetObject as Component;
235 if (!target)
236 return null;
237
238 var type = ResolveType(attr.ContextTypeName);
239 if (type == null)
240 return null;
241
242 Transform current = target.transform;
243 while (current)
244 {
245 var comp = current.GetComponent(type);
246 if (comp)
247 return comp;
248
249 current = current.parent;
250 }
251
252 return null;
253 }

255 private static Type ResolveType( string _typeName )
256 {
257 // Try by full name
258 var t = Type.GetType(_typeName);
259 if (t != null)
260 return t;
261
262 // Try by short name across loaded assemblies
263 foreach (var asm in AppDomain.CurrentDomain.GetAssemblies())
264 {
265 try
266 {
267 foreach (var assemblyType in asm.GetTypes())
268 {
269 if (assemblyType.Name == _typeName)
270 return assemblyType;
271 }
272 }
273 catch { /* ignore */ }
274 }
275 return null;
276 }

305 public static int ValidateOpenScenes( bool _logToConsole )
306 {
307 int violations = 0;
308
309 for (int i = 0; i < SceneManager.sceneCount; i++)
310 {
311 var s = SceneManager.GetSceneAt(i);
312 if (!s.isLoaded) continue;
313
314 foreach (var root in s.GetRootGameObjects())
315 {
316 var comps = root.GetComponentsInChildren<Component>(true);
317 foreach (var c in comps)
318 {
319 if (!c) continue;
320 if (PrefabUtility.IsPartOfPrefabAsset(c)) continue;
321
322 var so = new SerializedObject(c);
323 var sp = so.GetIterator();
324 if (sp.NextVisible(true))
325 {
326 do
327 {
328 if (sp.propertyType != SerializedPropertyType.ObjectReference) continue;
329
330 var field = GetFieldInfo(c.GetType(), sp.name);
331 if (field == null) continue;
332
333 var attr = (MandatoryExternalAttribute)Attribute.GetCustomAttribute(field, typeof(MandatoryExternalAttribute));
334 if (attr == null) continue;
335
336 if (sp.objectReferenceValue == null)
337 {
338 violations++;
339 if (_logToConsole && attr.LogErrorInConsole)
340 {
341 string compName = c.GetType().Name;
342 string fieldName = field.Name;
343
344 // Context hint
345 string ctx = string.IsNullOrEmpty(attr.ContextNameHint) ? "external context" : attr.ContextNameHint;
346 UiLog.LogError(
347 $"ExternalRequired violation: <{compName}.{fieldName}> is null. " +
348 $"This reference must be assigned in {ctx}.",
349 c
350 );
351 }
352 }
353 }
354 while (sp.NextVisible(false));
355 }
356 }
357 }
358 }
359
360 if (_logToConsole)
361 {
362 if (violations == 0)
363 UiLog.LogInternal("ExternalRequired: No violations found in open scenes.");

366 }
