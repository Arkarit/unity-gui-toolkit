53 public DoxygenRunner(string[] args, DoxygenThreadSafeOutput safeoutput, Action<int> callback)
54 {
55 if (DoxygenPath == null)
56 throw new FileNotFoundException("Doxygen Executable not found");
57
58 m_args = args;
59 m_output = safeoutput;
60 m_onCompleteCallBack = callback;
61 m_workingFolder = Path.GetDirectoryName(s_doxygenExecutablePath);
62 }
63
64 public void RunThreadedDoxygen()
65 {
66 Action<string> outputHandler = (string output) => UpdateOutputString(output);
67 int ReturnCode = EditorProcessHelper.Run(null, "doxygen", m_workingFolder, outputHandler, m_args);
68 m_output.WriteFullLog(m_doxyLog);
69 m_output.SetFinished();
70 m_onCompleteCallBack(ReturnCode);
71 }
72
73 private void UpdateOutputString(string output)
