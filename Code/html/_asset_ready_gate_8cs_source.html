92 public static bool IsReady(bool _includeBootstrap)
93 {
94 if (Application.isPlaying)
95 return true;
96
97 if (_includeBootstrap && !Bootstrap.IsInitialized)
98 return false;
99
100 if (m_allScriptableObjectsReady)
101 return true;
102
103 if (BuildPipeline.isBuildingPlayer)
104 return false;
105
106 // Editor must be calm first.
107 if (ImportBusy)
108 return false;
109
110 for (int i = 0; i < ScriptableObjectGuids.Length; i++)
111 {
112 string guid = ScriptableObjectGuids[i];
113 string path = AssetDatabase.GUIDToAssetPath(guid);
114
115 // If path cannot be resolved (and editor is calm), treat as missing, not pending.
116 if (string.IsNullOrEmpty(path))
117 continue;
118
119 // Folders are not assets to wait for.
120 if (AssetDatabase.IsValidFolder(path))
121 continue;
122
123 // If file is not on disk (and editor is calm), treat as missing, not pending.
124 if (!File.Exists(path))
125 continue;
126
127 // Importer is done when the main type is known.
128 Type mainType = AssetDatabase.GetMainAssetTypeAtPath(path);
129 if (mainType == null)
130 return false; // Still pending.
131 }
132
133 // All checked paths are ready (or irrelevant).
134 m_allScriptableObjectsReady = true;
135 return true;
136 }

146 public static void WhenReady
147 (
148 Action _callback,
149 bool _includeBootstrap = true,
150 int _quietFrames = 5,
151 int _maxFrames = 600 // 0 = no timeout
152 )
153 {
154 if (_callback == null)
155 return;
156
157 if (Application.isPlaying)
158 {
159 _callback();
160 return;
161 }
162
163 // Immediate fast path
164 if (IsReady(_includeBootstrap))
165 {
166 _callback();
167 return;
168 }
169
170 int countdown = _quietFrames;
171 int frames = 0;
172
173 UiLog.LogOnce("Begin waiting for Scriptable Objects to be available");
174 EditorApplication.update += Tick;
175 return;
176
177 void Tick()
178 {
179 // Abort if a build begins after scheduling.
180 if (BuildPipeline.isBuildingPlayer)
181 {
182 EditorApplication.update -= Tick;
183 UiLog.LogWarning("WhenReady aborted: build started.");
184 return;
185 }
186
187 // We need to check if the editor is in background;
188 // since it is slower in background, we often get a timeout
189 if (!EditorApplication.isFocused)
190 return;
191
192 // Timeout guard to avoid dangling subscriptions.
193 if (_maxFrames > 0 && ++frames > _maxFrames)
194 {
195 EditorApplication.update -= Tick;
196 UiLog.LogErrorOnce($"WhenReady(assets) timeout after {_maxFrames} frames. Caller: {DebugUtility.GetCallingClassAndMethod()}");
197 return;
198 }
199
200 // Not ready yet; reset quiet countdown.
201 if (!IsReady(_includeBootstrap))
202 {
203 countdown = _quietFrames;
204 return;
205 }
206
207 // Wait for the required number of consecutive quiet frames.
208 if (countdown-- > 0)
209 return;
210
211 EditorApplication.update -= Tick;
212 if (UiLog.LogOnce($"All scriptable objects became available after {frames} frames plus {_quietFrames} extra frames."))
213 {
214 if (s_scriptableObjectGuids != null)
215 {
216 string s = "Scriptable Objects ready:";
217
218 foreach (var guid in s_scriptableObjectGuids)
219 {
220 if (guid == null)
221 continue;
222
223 s += $"\n\t{AssetDatabase.GUIDToAssetPath(guid)}";
224 }
225
226 UiLog.LogVerbose(s);
227 }
228 }
229
230 try
231 {
232 _callback();
233 }
234 catch (Exception ex)
235 {
236 UiLog.LogError($"Exception in Callback: {ex}");
237 }
238 }
239 }

295 public static ScriptableObject EditorLoad( Type _type )
296 {
297 ThrowIfNotReady();

299 var foundGuids = AssetDatabase.FindAssets($"t:{_type.Name}");
300 if (foundGuids == null || foundGuids.Length == 0)

307 if (path.StartsWith("Assets/", StringComparison.Ordinal))

311 path = AssetDatabase.GUIDToAssetPath(foundGuids[0]);

323 public static ScriptableObject EditorLoadOrCreate( Type _type, out bool _wasCreated )
324 {
325 _wasCreated = false;
326 // We do not need a ThrowIfNotReady() here; EditorLoad() already enforces it.
327 var asset = EditorLoad(_type);
328 if (asset)
329 return asset;
330
331 // Hard guard: never create assets while building the player.
332 if (BuildPipeline.isBuildingPlayer)
333 {
334 UiLog.LogError($"Attempt to create scriptable object '{_type.Name}' during build process");
335 return null;
336 }
337
338 _wasCreated = true;
339 var assetPath = $"{AssetDir}{_type.Name}.asset";
340 EditorFileUtility.EnsureUnityFolderExists(System.IO.Path.GetDirectoryName(assetPath).Replace('\\', '/'));
341 var inst = ScriptableObject.CreateInstance(_type);
342 inst.name = _type.Name;
343 UiLog.LogInternal($"Create scriptable object instance '{inst.name}' of type '{_type.Name}' at '{assetPath}'");
344 AssetDatabase.CreateAsset(inst, assetPath);
345 AssetDatabase.SaveAssets();
346 AssetDatabase.ImportAsset(assetPath, ImportAssetOptions.ForceUpdate | ImportAssetOptions.ForceSynchronousImport);
347 return AssetDatabase.LoadAssetAtPath<ScriptableObject>(assetPath);
348 }
