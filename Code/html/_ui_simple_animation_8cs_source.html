597 private void AnimateSkew(float _normalizedTime)
598 {
599 bool animateX = m_skewCurveHorizontal != null && m_skewCurveHorizontal.length > 1;
600 bool animateY = m_skewCurveVertical != null && m_skewCurveVertical.length > 1;
601 var x = animateX ? Mathf.LerpUnclamped(m_skewMinHorizontal, m_skewMaxHorizontal, m_skewCurveHorizontal.Evaluate(_normalizedTime)) : 0;
602 var y = animateX ? Mathf.LerpUnclamped(m_skewMinVertical, m_skewMaxVertical, m_skewCurveVertical.Evaluate(_normalizedTime)) : 0;
603
604 if (animateX && animateY)
605 {
606 m_uiSkew.Angles = new Vector2(x, y);
607 return;
608 }
609
610 if (animateX)
611 {
612 m_uiSkew.AngleHorizontal = x;
613 return;
614 }
615
616 if (animateY)
617 m_uiSkew.AngleVertical = x;
618 }
