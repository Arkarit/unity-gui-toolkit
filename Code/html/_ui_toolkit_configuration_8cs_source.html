230 private void InitScenesByName()
231 {
232 m_scenesByName.Clear();
233 if (m_sceneReferences == null)
234 m_sceneReferences = new SceneReference[0];
235
236#if UNITY_EDITOR
237 // Always refresh scene references list, but only in editor mode
238 if (!Application.isPlaying)
239 {
240 var sceneReferences = BuildSettingsUtility.GetBuildSceneReferences();
241 if (sceneReferences != null)
242 {
243 bool shouldStore = sceneReferences.Length != m_sceneReferences.Length;
244 if (!shouldStore)
245 {
246 for (int i = 0; i < sceneReferences.Length; i++)
247 {
248 if (sceneReferences[i] != m_sceneReferences[i])
249 {
250 shouldStore = true;
251 break;
252 }
253 }
254 }
255
256 if (shouldStore)
257 {
258 m_sceneReferences = sceneReferences;
259 EditorUtility.SetDirty(this);
260 }
261 }
262 }
263#endif
264
265 foreach (var sceneReference in m_sceneReferences)
266 {
267 string name = Path.GetFileNameWithoutExtension(sceneReference.ScenePath);
268 if (m_scenesByName.ContainsKey(name))
269 {
270 UiLog.LogError($"Found non-unique scene name for '{name}'");
271 continue;
272 }
273 m_scenesByName.Add(name, sceneReference);
274 }
