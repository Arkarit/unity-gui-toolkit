209 private IEnumerator LoadAsyncScene( string _name, bool _show, bool _instant, Action<UiView> _whenLoaded )
210 {
211 // The Application loads the Scene in the background as the current Scene runs.
212 // This is particularly good for creating loading screens.
213 // You could also load the Scene by using sceneBuildIndex. In this case Scene2 has
214 // a sceneBuildIndex of 1 as shown in Build Settings.
215
216 yield return new WaitUntil(() => AssetReadyGate.Ready);
217
218 AsyncOperation asyncLoad = SceneManager.LoadSceneAsync(UiToolkitConfiguration.Instance.GetScenePath(_name), LoadSceneMode.Additive);
219
220 if (asyncLoad == null)
221 yield break;
222
223 asyncLoad.completed += ( AsyncOperation operation ) =>
224 {
225 Scene scene = SceneManager.GetSceneByName(_name);
226 if (scene == null)
227 return;
228
229 var roots = scene.GetRootGameObjects();
230 foreach (var root in roots)
231 {
232 UiView view = root.GetComponentInChildren<UiView>(true);
233 if (view)
234 {
235 SceneManager.MoveGameObjectToScene(view.gameObject, SceneManager.GetSceneAt(0));
236
237 view.gameObject.name = _name;
238 view.transform.SetParent(transform, false);
239 SetDefaultSceneVisibilities(root);
240 m_scenes[_name] = view;
241
242 if (_show)
243 view.ShowTopmost(_instant);
244
245 if (_whenLoaded != null)
246 _whenLoaded.Invoke(view);
247
248 break;
249 }
250 }
251
252 roots = scene.GetRootGameObjects();
253 foreach (var root in roots)
254 Destroy(root);
255
256 SceneManager.UnloadSceneAsync(scene);
257 };
258
259 yield return asyncLoad;
260
261 }

278 public void NavigationPop( int _skip = 0, bool _instant = false, Action _onFinishHide = null, Action _onFinishShow = null )
279 {
280 bool stackValid = m_stack.Count >= 1 + _skip;
281 if (!stackValid)
282 {
283 UiLog.LogError($"Attempting to pop {1 + _skip} from UiView stack, but stack contains only {m_stack.Count}");
284 return;
285 }
286 UiView currentShown = m_stack.Pop();
287 currentShown.SetStackAnimationType(m_stackAnimationType, true, m_stackPushedOutCurve);
288 currentShown.Hide(_instant, _onFinishHide);
289
290 for (int i = 0; i < _skip; i++)
291 m_stack.Pop();
292
293 if (m_stack.Count > 0)
294 {
295 UiView nextShown = m_stack.Peek();
296 nextShown.SetStackAnimationType(m_stackAnimationType, false, m_stackMovedInCurve);
297 nextShown.ShowTopmost(_instant, _onFinishShow);
298 }
299 }

547 private void SetActivenessForTagged( bool _active )
548 {
549 if (_active)
550 {
551 for (int i = 0; i < m_tagsToDisableWhenFullScreenView.Length; i++)
552 {
553 string tag = m_tagsToDisableWhenFullScreenView[i];
554 if (string.IsNullOrEmpty(tag))
555 continue;
556
557 var gameObjects = GameObject.FindGameObjectsWithTag(tag);
558 foreach (var go in gameObjects)
559 {
560 go.SetActive(false);
561 m_hiddenTaggedGameObjects.Add(go);
562 }
563 }
564
565 return;
566 }
567
568 foreach (var go in m_hiddenTaggedGameObjects)
569 if (go)
570 go.SetActive(true);
571
572 m_hiddenTaggedGameObjects.Clear();
573 }

587 private void SetPlaneDistancesBySiblingIndex()
588 {
589 var layers = EnumHelper.GetValues<EUiLayerDefinition>();
590 foreach (var layer in layers)
591 {
592 int layercount = 0;
593 foreach (Transform childTransform in transform)
594 {
595 UiView view = childTransform.GetComponent<UiView>();
596 if (!view || view.Layer != layer)
597 continue;
598
599 int planeIndex = (int)layer - layercount;
600 float planeDistance = LayerDistance * (float)planeIndex;
601 view.InitView(RenderMode, Camera, planeDistance, (int)EUiLayerDefinition.Back - planeIndex);
602
603 layercount++;
604 }
605 }

608 private class PlaneDistanceComparer : IComparer<Transform>
609 {
610 // Call CaseInsensitiveComparer.Compare with the parameters reversed.
611 public int Compare( Transform a, Transform b )
612 {
613 Canvas canvasA = a.GetComponent<Canvas>();
614 Canvas canvasB = b.GetComponent<Canvas>();
615 if (canvasA == null)
616 return canvasB != null ? -1 : 0;
617 if (canvasB == null)
618 return 1;
619 if (canvasA.planeDistance == canvasB.planeDistance)
620 return 0;
621 return canvasA.planeDistance < canvasB.planeDistance ? 1 : -1;
622 }
623 }
624
625 private void SetSiblingIndicesByPlaneDistances()
626 {
627 Transform[] children = transform.GetChildrenArray();
628 PlaneDistanceComparer comparer = new PlaneDistanceComparer();
629 Array.Sort(children, comparer);
630 for (int i = 0; i < children.Length; i++)
631 children[i].SetSiblingIndex(i);
632 }

773 if (!EditorGameObjectUtility.IsInPrefabEditingMode && m_renderMode == RenderMode.ScreenSpaceCamera || m_renderMode == RenderMode.WorldSpace)
774 {
775 Camera[] cameras = Camera.allCameras;
776 Camera uiCamera = GetComponent<Camera>();
777 if (uiCamera == null)
778 throw new Exception($"UiMain GameObject '{gameObject.name}' needs an attached camera!");
779
780 int uiLayer = 1 << LayerMask.NameToLayer("UI");
781 if ((uiCamera.cullingMask & uiLayer) == 0)
782 {
783 UiLog.LogError($"UiMain camera on GameObject '{gameObject.name}' needs the culling mask 'UI' flag set!");
784 }
785
786 float uiCameraDepth = uiCamera.depth;
787
788 foreach (var camera in cameras)
789 {
790 // weird: As of a specific Unity version between 2019.3 and 2020.3, camera == uiCamera is never true in prefab editing mode.
791 // Thus, EditorUiUtility.IsInPrefabEditingMode guard above.
792 if (camera == uiCamera || camera.targetTexture != null)
793 continue;
794
795 if (camera.depth >= uiCameraDepth)
796 {
797 UiLog.LogError($"UI Camera needs highest depth. Camera depth {camera.depth} detected on camera '{camera.gameObject.name}', which is >= Ui camera depth ({uiCameraDepth})");
798 }
799 }
800 }
