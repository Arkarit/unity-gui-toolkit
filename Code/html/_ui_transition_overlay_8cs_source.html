168 group.alpha = 0f; // start invisible

186 blackRt.anchorMax = Vector2.one;
187 blackRt.offsetMin = Vector2.zero;
188 blackRt.offsetMax = Vector2.zero;

190 Image blackImage = blackGo.GetComponent<Image>();
