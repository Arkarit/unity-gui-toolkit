159 private static string GetCallsiteKey()
160 {
161 var st = new StackTrace(true);
162
163 for (int i = 0; i < st.FrameCount; i++)
164 {
165 var f = st.GetFrame(i);
166 var m = f.GetMethod();
167 if (m == null)
168 continue;
169
170 var declaringType = m.DeclaringType;
171 if (declaringType == typeof(UiLog))
172 continue;
173
174 string file = f.GetFileName();
175 int line = f.GetFileLineNumber();
176
177 if (!string.IsNullOrEmpty(file) && line > 0)
178 return file + ":" + line;
179
180 // Fallback when PDBs or file info are not available (e.g., some player builds)
181 string typeName = declaringType != null ? declaringType.FullName : "<UnknownType>";
182 string methodName = m is MethodInfo mi ? mi.Name : m.Name;
183 int il = f.GetILOffset();
184 return typeName + "." + methodName + "#" + il;
