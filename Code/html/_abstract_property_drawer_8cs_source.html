481 protected void CollectChildPropertiesSorted(SerializedProperty _property, List<SerializedProperty> list)
482 {
483 var enumerator = _property.Copy().GetEnumerator();
484 int depth = _property.depth;
485
486 while (enumerator.MoveNext())
487 {
488 var property = enumerator.Current as SerializedProperty;
489 if (property == null || property.depth > depth + 1)
490 continue;
491
492 list.Add(property.Copy());
493 }
494
495 list.Sort((a, b) => a.displayName.CompareTo(b.displayName));
496 }
