122 private string AddSkin(string _name, string _copyFromName)
123 {
124 if (m_thisUiStyleConfig.SkinNames.Contains(_name))
125 return string.Empty;
126
127 var newSkin = new UiSkin(m_thisUiStyleConfig, _name);
128
129 UiSkin copyFrom = null;
130 if (!string.IsNullOrEmpty(_copyFromName))
131 {
132 foreach (var skin in m_thisUiStyleConfig.Skins)
133 {
134 if (skin.Name == _copyFromName)
135 {
136 copyFrom = skin;
137 break;
138 }
139 }
140 }
141
142 if (copyFrom != null)
143 {
144 foreach (var style in copyFrom.Styles)
145 {
146 var newStyle = style.DeepClone();
147 newSkin.Styles.Add(newStyle);
148 }
149 }
150
151 m_skinsProp.arraySize += 1;
152 m_skinsProp.GetArrayElementAtIndex(m_skinsProp.arraySize - 1).boxedValue = newSkin;
153 serializedObject.ApplyModifiedProperties();
154 UiStyleConfig.SetDirty(m_thisUiStyleConfig);
155 return _name;
