55 private void CollectDependenciesFromSceneObject()
56 {
57 var go = Object as GameObject;
58 if (!go)
59 {
60 if (Object is Component comp) go = comp.gameObject;
61 if (!go) return;
62 }
63
64 // 1) Children are structural dependencies (deleting parent deletes children)
65 var t = go.transform;
66 for (int i = 0; i < t.childCount; ++i)
67 {
68 var childGo = t.GetChild(i).gameObject;
69 var childNode = Tree.GetOrCreateNode(childGo);
70 Link(this, childNode);
71 }
72
73 // 2) Component object-references (assets and stage objects)
74 foreach (var c in go.GetComponents<Component>())
75 {
76 if (!c) continue;
77
78 try
79 {
80 using (var so = new SerializedObject(c))
81 {
82 var sp = so.GetIterator();
83 while (sp.NextVisible(true))
84 {
85 if (sp.propertyType != SerializedPropertyType.ObjectReference)
86 continue;
87
88 var refObj = sp.objectReferenceValue;
89 if (!refObj) continue;
90
91 // Asset reference?
92 if (AssetDatabase.Contains(refObj))
93 {
94 var path = AssetDatabase.GetAssetPath(refObj);
95 if (string.IsNullOrEmpty(path)) continue;
96 if (path.EndsWith(".cs", StringComparison.OrdinalIgnoreCase)) continue; // ignore scripts
97
98 // Normalize subassets -> main asset
99 var main = AssetDatabase.LoadMainAssetAtPath(path);
100 if (!main) continue;
101
102 var depNode = Tree.GetOrCreateNode(main);
103 Link(this, depNode);
104 }
105 else
106 {
107 // Stage-side reference (component or GO)
108 var refGo = (refObj as GameObject) ?? (refObj as Component)?.gameObject;
109 if (!refGo) continue;
110
111 var depNode = Tree.GetOrCreateNode(refGo);
112 Link(this, depNode);
113 }
114 }
115 }
116 }
117 catch
118 {
119 // non-serializable editor components etc. – ignore in MVP
120 }
121 }
122 }

124 private void CollectDependenciesFromAsset()
125 {
126 if (string.IsNullOrEmpty(Path)) return;
127
128 // Unity's cached dependency listing. Always normalize to main asset.
129 var deps = AssetDatabase.GetDependencies(Path, true);
130 foreach (var depPath in deps)
131 {
132 if (string.Equals(depPath, Path, StringComparison.Ordinal))
133 continue; // skip self
134
135 var main = AssetDatabase.LoadMainAssetAtPath(depPath);
136 if (!main) continue;
137
138 var ext = System.IO.Path.GetExtension(depPath);
139 if (string.Equals(ext, ".cs", StringComparison.OrdinalIgnoreCase))
140 continue; // ignore scripts
141
142 var depNode = Tree.GetOrCreateNode(main);
143 Link(this, depNode);
144 }
145 }

160 private void SetProperties( Object _object )
161 {
162 // Normalize components to owning GameObject

164 _object = comp.gameObject;

166 Name = _object.name;

176 IsScript |= Path.EndsWith(".cs", StringComparison.OrdinalIgnoreCase);
177 IsInResources = Path.IndexOf("/Resources/", StringComparison.OrdinalIgnoreCase) >= 0;
178 IsInPackages = Path.StartsWith("Packages", StringComparison.OrdinalIgnoreCase);

180 }
