89 protected void DisplayBlendingHelper()
90 {
91 var srcBlendProp = FindProperty(PropNameSrcBlend, MaterialProperties);
92 BlendMode src = (BlendMode)srcBlendProp.floatValue;
93 var dstBlendProp = FindProperty(PropNameDstBlend, MaterialProperties);
94 BlendMode dst = (BlendMode)dstBlendProp.floatValue;
95
96 var bm = CommonBlendModes.Indeterminate;
97
98 if (src == BlendMode.SrcAlpha && dst == BlendMode.OneMinusSrcAlpha)
99 bm = CommonBlendModes.Transparent;
100 else if (src == BlendMode.One && dst == BlendMode.OneMinusSrcAlpha)
101 bm = CommonBlendModes.TransparentPremultiplied;
102 else if (src == BlendMode.One && dst == BlendMode.One)
103 bm = CommonBlendModes.Additive;
104 else if (src == BlendMode.OneMinusDstColor && dst == BlendMode.One)
105 bm = CommonBlendModes.AdditiveSoft;
106 else if (src == BlendMode.DstColor && dst == BlendMode.Zero)
107 bm = CommonBlendModes.Multiplicative;
108 else if (src == BlendMode.DstColor && dst == BlendMode.SrcColor)
109 bm = CommonBlendModes.MultiplicativeX2;
110
111 SmallSpace();
112 bm = (CommonBlendModes)EditorGUILayout.EnumPopup("Set Source/Destination by common blend mode:", bm);
113
114 switch (bm)
115 {
116 default:
117 case CommonBlendModes.Indeterminate:
118 break;
119 case CommonBlendModes.Transparent:
120 srcBlendProp.floatValue = (float)BlendMode.SrcAlpha;
121 dstBlendProp.floatValue = (float)BlendMode.OneMinusSrcAlpha;
122 break;
123 case CommonBlendModes.TransparentPremultiplied:
124 srcBlendProp.floatValue = (float)BlendMode.One;
125 dstBlendProp.floatValue = (float)BlendMode.OneMinusSrcAlpha;
126 break;
127 case CommonBlendModes.Additive:
128 srcBlendProp.floatValue = (float)BlendMode.One;
129 dstBlendProp.floatValue = (float)BlendMode.One;
130 break;
131 case CommonBlendModes.AdditiveSoft:
132 srcBlendProp.floatValue = (float)BlendMode.OneMinusDstColor;
133 dstBlendProp.floatValue = (float)BlendMode.One;
134 break;
135 case CommonBlendModes.Multiplicative:
136 srcBlendProp.floatValue = (float)BlendMode.DstColor;
137 dstBlendProp.floatValue = (float)BlendMode.Zero;
138 break;
139 case CommonBlendModes.MultiplicativeX2:
140 srcBlendProp.floatValue = (float)BlendMode.DstColor;
141 dstBlendProp.floatValue = (float)BlendMode.SrcColor;
142 break;
143 }
