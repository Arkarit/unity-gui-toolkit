94 public override void ModifyMesh( VertexHelper _vertexHelper )
95 {
96 if (!IsActive())
97 return;
98
99 _vertexHelper.GetUIVertexStream(s_verts);
100
101 Bounding = UiMeshModifierUtility.GetBounds(s_verts);
102
103 Prepare();
104
105 Vector2 size = IsAbsolute ? Vector2.one : Bounding.size;
106
107 Vector2 tl = TopLeft * size;
108 Vector2 bl = BottomLeft * size;
109 Vector2 tr = TopRight * size;
110 Vector2 br = BottomRight * size;
111
112 bool mirrorHorizontal = m_mirrorAxisFlags.IsFlagSet(EAxis2DFlags.Horizontal);
113 bool mirrorVertical = m_mirrorAxisFlags.IsFlagSet(EAxis2DFlags.Vertical);
114
115 Vector2 mirrorVec = new Vector2(mirrorHorizontal ? -1 : 1, mirrorVertical ? -1 : 1);
116
117 if (mirrorHorizontal)
118 {
119 UiMathUtility.Swap( ref tl, ref tr );
120 UiMathUtility.Swap( ref bl, ref br );
121 }
122
123 if (mirrorVertical)
124 {
125 UiMathUtility.Swap( ref tl, ref bl );
126 UiMathUtility.Swap( ref tr, ref br );
127 }
128
129 for (int i = 0; i < _vertexHelper.currentVertCount; ++i)
130 {
131 _vertexHelper.PopulateUIVertex(ref s_vertex, i);
132
133 Vector2 pointNormalized = s_vertex.position.GetNormalizedPointInRect(Bounding);
134 if (IsNaN(pointNormalized))
135 continue;
136
137 Vector2 point = s_vertex.position.Xy() + UiMathUtility.Lerp4P(ref tl, ref tr, ref bl, ref br, ref pointNormalized, false, true) * mirrorVec;
138 s_vertex.position = new Vector3(point.x, point.y, s_vertex.position.z);
139
140 _vertexHelper.SetUIVertex(s_vertex, i);
141 }
142 }
