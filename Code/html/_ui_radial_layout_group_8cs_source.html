97 private void CalculateRadial()
98 {
99 m_Tracker.Clear();
100
101 int childCount = ChildCount;
102 if (childCount == 0)
103 return;
104
105 float topAngleOffset = 0, angleIncrement = 0;
106 List<float> angleIncrements = null;
107
108 switch(m_mode)
109 {
110 case Mode.FixedOverallAngle:
111 topAngleOffset = -m_angle0 - 90;
112 angleIncrement = ((m_angle1 - m_angle0)) / (childCount - 1);
113 break;
114 case Mode.PerElementAngle:
115 angleIncrements = ElementAngles;
116 float sum = 0;
117 for (int i = 0; i<angleIncrements.Count; i++)
118 sum += angleIncrements[i];
119 topAngleOffset = sum * 0.5f - 90.0f;
120 break;
121 default:
122 case Mode.FixedElementAngle:
123 topAngleOffset = m_angle0 * (childCount - 1) * 0.5f - 90.0f;
124 angleIncrement = m_angle0;
125 break;
126 }
127
128 float runningAngle = m_angleOffset - topAngleOffset;
129
130 float z = 0;
131 if (m_useZIncrement && childCount > 1)
132 {
133 z = - m_zIncrement * childCount / 2;
134 }
135
136 int angleIncrementsIdx = 0;
137 for (int i = 0; i < transform.childCount; i++)
138 {
139 RectTransform child = (RectTransform)transform.GetChild(i);
140 if (!child.gameObject.activeSelf)
141 continue;
142
143 if (child != null)
144 {
145 m_Tracker.Add( this, child, GetDrivenTransformProperties() );
146
147 float angle = runningAngle;
148 if (m_mode == Mode.PerElementAngle)
149 angle += angleIncrements[angleIncrementsIdx] * 0.5f;
150
151 Vector3 vPos = new Vector3(Mathf.Cos(angle * Mathf.Deg2Rad) * m_xFactor, Mathf.Sin(angle * Mathf.Deg2Rad), 0);
152 child.localPosition = vPos * m_radius;
153
154 if (m_useZIncrement)
155 {
156 child.localPosition += new Vector3(0,0, z);
157 z += m_zIncrement;
158 }
159
160 if (m_rotateElements)
161 {
162 child.localRotation = Quaternion.AngleAxis(angle + m_elementAngleOffset, Vector3.forward);
163 }
164 else if (m_childRotationChanged)
165 {
166 child.localRotation = Quaternion.identity;
167 }
168
169 if (m_mode == Mode.PerElementAngle)
170 runningAngle += angleIncrements[angleIncrementsIdx++];
171 else
172 runningAngle += angleIncrement;
173 }
174 }
175
176 m_childRotationChanged = false;
177 }
